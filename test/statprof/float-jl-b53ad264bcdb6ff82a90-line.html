135 function convert(::Type{Float16}, val::Float32)
136 f = reinterpret(UInt32, val)
137 i = (f >> 23) & 0x1ff + 1
138 sh = shifttable[i]
139 f &= 0x007fffff
140 h::UInt16 = basetable[i] + (f >> sh)
141 # round
142 # NOTE: we maybe should ignore NaNs here, but the payload is
143 # getting truncated anyway so "rounding" it might not matter
144 nextbit = (f >> (sh-1)) & 1
145 if nextbit != 0
146 # Round halfway to even or check lower bits
147 if h&1 == 1 || (f & ((1<<(sh-1))-1)) != 0
148 h += 1
149 end
150 end

154 function convert(::Type{Float32}, val::Float16)
155 local ival::UInt32 = reinterpret(UInt16, val)
156 local sign::UInt32 = (ival & 0x8000) >> 15
157 local exp::UInt32 = (ival & 0x7c00) >> 10
158 local sig::UInt32 = (ival & 0x3ff) >> 0
159 local ret::UInt32
160
161 if exp == 0
162 if sig == 0
163 sign = sign << 31
164 ret = sign | exp | sig
165 else
166 n_bit = 1
167 bit = 0x0200
168 while (bit & sig) == 0
169 n_bit = n_bit + 1
170 bit = bit >> 1
171 end
172 sign = sign << 31
173 exp = (-14 - n_bit + 127) << 23
174 sig = ((sig & (~bit)) << n_bit) << (23 - 10)
175 ret = sign | exp | sig
176 end
177 elseif exp == 0x1f
178 if sig == 0 # Inf
179 if sign == 0
180 ret = 0x7f800000
181 else
182 ret = 0xff800000
183 end
184 else # NaN
185 ret = 0x7fc00000 | (sign<<31)
186 end
187 else
188 sign = sign << 31
189 exp = (exp - 15 + 127) << 23
190 sig = sig << (23 - 10)
191 ret = sign | exp | sig
192 end
193 return reinterpret(Float32, ret)
194 end

196 # Float32 -> Float16 algorithm from:
197 # "Fast Half Float Conversion" by Jeroen van der Zijp
198 # ftp://ftp.fox-toolkit.org/pub/fasthalffloatconversion.pdf

200 const basetable = Vector{UInt16}(512)
201 const shifttable = Vector{UInt8}(512)
202
203 for i = 0:255
204 e = i - 127
205 if e < -24 # Very small numbers map to zero
206 basetable[i|0x000+1] = 0x0000
207 basetable[i|0x100+1] = 0x8000
208 shifttable[i|0x000+1] = 24
209 shifttable[i|0x100+1] = 24
210 elseif e < -14 # Small numbers map to denorms
211 basetable[i|0x000+1] = (0x0400>>(-e-14))
212 basetable[i|0x100+1] = (0x0400>>(-e-14)) | 0x8000
213 shifttable[i|0x000+1] = -e-1
214 shifttable[i|0x100+1] = -e-1
215 elseif e <= 15 # Normal numbers just lose precision
216 basetable[i|0x000+1] = ((e+15)<<10)
217 basetable[i|0x100+1] = ((e+15)<<10) | 0x8000
218 shifttable[i|0x000+1] = 13
219 shifttable[i|0x100+1] = 13
220 elseif e < 128 # Large numbers map to Infinity
221 basetable[i|0x000+1] = 0x7C00
222 basetable[i|0x100+1] = 0xFC00
223 shifttable[i|0x000+1] = 24
224 shifttable[i|0x100+1] = 24
225 else # Infinity and NaN's stay Infinity and NaN's
226 basetable[i|0x000+1] = 0x7C00
227 basetable[i|0x100+1] = 0xFC00
228 shifttable[i|0x000+1] = 13
229 shifttable[i|0x100+1] = 13
230 end

459 # Exact Float (Tf) vs Integer (Ti) comparisons
460 # Assumes:
461 # - typemax(Ti) == 2^n-1
462 # - typemax(Ti) can't be exactly represented by Tf:
463 # => Tf(typemax(Ti)) == 2^n or Inf
464 # - typemin(Ti) can be exactly represented by Tf
465 #
466 # 1. convert y::Ti to float fy::Tf
467 # 2. perform Tf comparison x vs fy
468 # 3. if x == fy, check if (1) resulted in rounding:
469 # a. convert fy back to Ti and compare with original y
470 # b. unsafe_convert undefined behaviour if fy == Tf(typemax(Ti))
471 # (but consequently x == fy > y)
472 for Ti in (Int64,UInt64,Int128,UInt128)
473 for Tf in (Float32,Float64)
474 @eval begin
475 function ==(x::$Tf, y::$Ti)
476 fy = ($Tf)(y)
477 (x == fy) & (fy != $(Tf(typemax(Ti)))) & (y == unsafe_trunc($Ti,fy))
478 end
479 ==(y::$Ti, x::$Tf) = x==y

484 end

581 nextfloat(x::AbstractFloat, n::Integer)
582
583 The result of `n` iterative applications of `nextfloat` to `x` if `n >= 0`, or `-n`
584 applications of `prevfloat` if `n < 0`.
585 """
586 function nextfloat(f::Union{Float16,Float32,Float64}, d::Integer)
587 F = typeof(f)
588 fumax = reinterpret(Unsigned, F(Inf))
589 U = typeof(fumax)
590
591 isnan(f) && return f
592 fi = reinterpret(Signed, f)
593 fneg = fi < 0
594 fu = unsigned(fi & typemax(fi))
595
596 dneg = d < 0
597 da = uabs(d)
598 if da > typemax(U)
599 fneg = dneg
600 fu = fumax
601 else
602 du = da % U
603 if fneg √¢¬ä¬ª dneg
604 if du > fu
605 fu = min(fumax, du - fu)
606 fneg = !fneg
607 else
608 fu = fu - du
609 end
610 else
611 if fumax - fu < du
612 fu = fumax
613 else
614 fu = fu + du
615 end
616 end
617 end
618 if fneg
619 fu |= sign_mask(F)
620 end
621 reinterpret(F, fu)
622 end

742 """
743 eps(x::AbstractFloat)
744
745 Returns the *unit in last place* (ulp) of `x`. This is the distance between consecutive
746 representable floating point values at `x`. In most cases, if the distance on either side
747 of `x` is different, then the larger of the two is taken, that is
748
749 eps(x) == max(x-prevfloat(x), nextfloat(x)-x)
750
751 The exceptions to this rule are the smallest and largest finite values
752 (e.g. `nextfloat(-Inf)` and `prevfloat(Inf)` for [`Float64`](@ref)), which round to the
753 smaller of the values.
754
755 The rationale for this behavior is that `eps` bounds the floating point rounding
756 error. Under the default `RoundNearest` rounding mode, if ``y`` is a real number and ``x``
757 is the nearest floating point number to ``y``, then
758
759 ```math
760 |y-x| \\leq \\operatorname{eps}(x)/2.
761 ```
762
763 ```jldoctest
764 julia> eps(1.0)
765 2.220446049250313e-16
766
767 julia> eps(prevfloat(2.0))
768 2.220446049250313e-16
769
770 julia> eps(2.0)
771 4.440892098500626e-16
772
773 julia> x = prevfloat(Inf) # largest finite Float64
774 1.7976931348623157e308
775
776 julia> x + eps(x)/2 # rounds up
777 Inf
778
779 julia> x + prevfloat(eps(x)/2) # rounds down
780 1.7976931348623157e308
781 ```
782 """
