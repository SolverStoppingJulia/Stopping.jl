28 # binary GCD (aka Stein's) algorithm
29 # about 1.7x (2.1x) faster for random Int64s (Int128s)
30 function gcd(a::T, b::T) where T<:Union{Int64,UInt64,Int128,UInt128}
31 a == 0 && return abs(b)
32 b == 0 && return abs(a)
33 za = trailing_zeros(a)
34 zb = trailing_zeros(b)
35 k = min(za, zb)
36 u = unsigned(abs(a >> za))
37 v = unsigned(abs(b >> zb))
38 while u != v
39 if u > v
40 u, v = v, u
41 end
42 v -= u
43 v >>= trailing_zeros(v)
44 end
45 r = u << k
46 # T(r) would throw InexactError; we want OverflowError instead
47 r > typemax(T) && throw(OverflowError())
48 r % T
49 end

86 gcdx(x,y)
87
88 Computes the greatest common (positive) divisor of `x` and `y` and their BÃ©zout
89 coefficients, i.e. the integer coefficients `u` and `v` that satisfy
90 ``ux+vy = d = gcd(x,y)``. ``gcdx(x,y)`` returns ``(d,u,v)``.
91
92 # Examples
93 ```jldoctest
94 julia> gcdx(12, 42)
95 (6, -3, 1)
96
97 julia> gcdx(240, 46)
98 (2, -9, 47)
99 ```
100
101 !!! note
102 BÃ©zout coefficients are *not* uniquely defined. `gcdx` returns the minimal
103 BÃ©zout coefficients that are computed by the extended Euclidean algorithm.
104 (Ref: D. Knuth, TAoCP, 2/e, p. 325, Algorithm X.)
105 For signed integers, these coefficients `u` and `v` are minimal in
106 the sense that ``|u| < |y/d|`` and ``|v| < |x/d|``. Furthermore,
107 the signs of `u` and `v` are chosen so that `d` is positive.
108 For unsigned integers, the coefficients `u` and `v` might be near
109 their `typemax`, and the identity then holds only via the unsigned
110 integers' modulo arithmetic.
111 """
112 function gcdx(a::T, b::T) where T<:Integer
113 # a0, b0 = a, b
114 s0, s1 = oneunit(T), zero(T)
115 t0, t1 = s1, s0
116 # The loop invariant is: s0*a0 + t0*b0 == a
117 while b != 0
118 q = div(a, b)
119 a, b = b, rem(a, b)
120 s0, s1 = s1, s0 - q*s1
121 t0, t1 = t1, t0 - q*t1
122 end
123 a < 0 ? (-a, -s0, -t0) : (a, s0, t0)
124 end

249 function powermod(x::Integer, p::Integer, m::T) where T<:Integer

261 p -= t
262 end

757 function isqrt(x::Union{Int64,UInt64,Int128,UInt128})
758 x==0 && return x
759 s = oftype(x, trunc(sqrt(x)))
760 # fix with a Newton iteration, since conversion to float discards
761 # too many bits.
762 s = (s + div(x,s)) >> 1
763 s*s > x ? s-1 : s
764 end

766 function factorial(n::Integer)
767 n < 0 && throw(DomainError())
768 local f::typeof(n*n), i::typeof(n*n)
769 f = 1
770 for i = 2:n
771 f *= i
772 end
773 return f

776 """
777 binomial(n, k)
778
779 Number of ways to choose `k` out of `n` items.
780
781 # Example
782 ```jldoctest
783 julia> binomial(5, 3)
784 10
785
786 julia> factorial(5) Ã· (factorial(5-3) * factorial(3))

788 ```
789 """
790 function binomial(n::T, k::T) where T<:Integer
791 k < 0 && return zero(T)
792 sgn = one(T)
793 if n < 0
794 n = -n + k -1
795 if isodd(k)
796 sgn = -sgn
797 end
798 end
799 k > n && return zero(T)
800 (k == 0 || k == n) && return sgn
801 k == 1 && return sgn*n
802 if k > (n>>1)
803 k = (n - k)
804 end
805 x::T = nn = n - k + 1
806 nn += 1
807 rr = 2
808 while rr <= k
809 xt = div(widemul(x, nn), rr)
810 x = xt
811 x == xt || throw(OverflowError())
812 rr += 1
813 nn += 1
814 end
815 convert(T, copysign(x, sgn))
816 end
