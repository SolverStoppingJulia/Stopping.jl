233
234 Generates a sequence of `if ... elseif ... else ... end` statements. For example:

261 function inlineanonymous(ex::Expr, val)
262 if ex.head != :->
263 throw(ArgumentError("not an anonymous function"))
264 end
265 if !isa(ex.args[1], Symbol)
266 throw(ArgumentError("not a single-argument anonymous function"))
267 end
268 sym = ex.args[1]
269 ex = ex.args[2]
270 exout = lreplace(ex, sym, val)
271 exout = poplinenum(exout)
272 exprresolve(exout)
273 end

298 function lreplace!(str::AbstractString, r::LReplace)

300 pat = r.pat_str
301 j = start(pat)
302 matching = false
303 while !done(str, i)
304 cstr, i = next(str, i)
305 if !matching
306 if cstr != '_' || done(str, i)
307 continue
308 end
309 istart = i
310 cstr, i = next(str, i)
311 end
312 if !done(pat, j)
313 cr, j = next(pat, j)
314 if cstr == cr
315 matching = true

318 j = start(pat)

320 continue

322 end
323 if matching && done(pat, j)
324 if done(str, i) || next(str, i)[1] == '_'

395 function exprresolve(ex::Expr)
396 for i = 1:length(ex.args)
397 ex.args[i] = exprresolve(ex.args[i])
398 end
399 # Handle simple arithmetic
400 can_eval, result = exprresolve_arith(ex)
401 if can_eval
402 return result
403 elseif ex.head == :call && (ex.args[1] == :+ || ex.args[1] == :-) && length(ex.args) == 3 && ex.args[3] == 0
404 # simplify x+0 and x-0
405 return ex.args[2]
406 end
407 # Resolve array references
408 if ex.head == :ref && isa(ex.args[1], Array)
409 for i = 2:length(ex.args)
410 if !isa(ex.args[i], Real)
411 return ex
412 end
413 end
414 return ex.args[1][ex.args[2:end]...]
415 end
416 # Resolve conditionals
417 if ex.head == :if
418 can_eval, tf = exprresolve_conditional(ex.args[1])
419 if can_eval
420 ex = tf?ex.args[2]:ex.args[3]
421 end
422 end
423 ex
