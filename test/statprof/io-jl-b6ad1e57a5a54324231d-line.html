172 """
173 join(io::IO, strings, delim, [last])
174
175 Join an array of `strings` into a single string, inserting the given delimiter between
176 adjacent strings. If `last` is given, it will be used instead of `delim` between the last
177 two strings. For example,
178
179 ```jldoctest
180 julia> join(["apples", "bananas", "pineapples"], ", ", " and ")
181 "apples, bananas and pineapples"
182 ```
183
184 `strings` can be any iterable over elements `x` which are convertible to strings
185 via `print(io::IOBuffer, x)`. `strings` will be printed to `io`.
186 """
187 function join(io::IO, strings, delim, last)
188 i = start(strings)
189 if done(strings,i)
190 return
191 end
192 str, i = next(strings,i)
193 print(io, str)
194 is_done = done(strings,i)
195 while !is_done
196 str, i = next(strings,i)
197 is_done = done(strings,i)
198 print(io, is_done ? last : delim)
199 print(io, str)
200 end
201 end

225 """
226 escape_string([io,] str::AbstractString[, esc::AbstractString]) -> AbstractString
227
228 General escaping of traditional C and Unicode escape sequences.
229 Any characters in `esc` are also escaped (with a backslash).
230 See also [`unescape_string`](@ref).
231 """
232 function escape_string(io, s::AbstractString, esc::AbstractString)
233 i = start(s)
234 while !done(s,i)
235 c, j = next(s,i)
236 c == '\0' ? print(io, escape_nul(s,j)) :
237 c == '\e' ? print(io, "\\e") :
238 c == '\\' ? print(io, "\\\\") :
239 c in esc ? print(io, '\\', c) :
240 '\a' <= c <= '\r' ? print(io, '\\', "abtnvfr"[Int(c)-6]) :
241 isprint(c) ? print(io, c) :
242 c <= '\x7f' ? print(io, "\\x", hex(c, 2)) :
243 c <= '\uffff' ? print(io, "\\u", hex(c, need_full_hex(s,j) ? 4 : 2)) :
244 print(io, "\\U", hex(c, need_full_hex(s,j) ? 8 : 4))
245 i = j
246 end
247 end

278 """
279 unescape_string([io,] s::AbstractString) -> AbstractString
280
281 General unescaping of traditional C and Unicode escape sequences. Reverse of
282 [`escape_string`](@ref).
283 """
284 function unescape_string(io, s::AbstractString)
285 i = start(s)
286 while !done(s,i)
287 c, i = next(s,i)
288 if !done(s,i) && c == '\\'
289 c, i = next(s,i)
290 if c == 'x' || c == 'u' || c == 'U'
291 n = k = 0
292 m = c == 'x' ? 2 :
293 c == 'u' ? 4 : 8
294 while (k+=1) <= m && !done(s,i)
295 c, j = next(s,i)
296 n = '0' <= c <= '9' ? n<<4 + c-'0' :
297 'a' <= c <= 'f' ? n<<4 + c-'a'+10 :
298 'A' <= c <= 'F' ? n<<4 + c-'A'+10 : break
299 i = j
300 end
301 if k == 1
302 throw(ArgumentError("invalid $(m == 2 ? "hex (\\x)" :
303 "unicode (\\u)") escape sequence used in $(repr(s))"))
304 end
305 if m == 2 # \x escape sequence
306 write(io, UInt8(n))
307 else
308 print(io, Char(n))
309 end
310 elseif '0' <= c <= '7'
311 k = 1
312 n = c-'0'
313 while (k+=1) <= 3 && !done(s,i)
314 c, j = next(s,i)
315 n = ('0' <= c <= '7') ? n<<3 + c-'0' : break
316 i = j
317 end
318 if n > 255
319 throw(ArgumentError("octal escape sequence out of range"))
320 end
321 write(io, UInt8(n))
322 else
323 print(io, c == 'a' ? '\a' :
324 c == 'b' ? '\b' :
325 c == 't' ? '\t' :
326 c == 'n' ? '\n' :
327 c == 'v' ? '\v' :
328 c == 'f' ? '\f' :
329 c == 'r' ? '\r' :
330 c == 'e' ? '\e' : c)
331 end
332 else
333 print(io, c)
334 end
335 end
336 end

353 function indentation(str::AbstractString; tabwidth=8)
354 count = 0
355 for ch in str
356 if ch == ' '
357 count += 1
358 elseif ch == '\t'
359 count = div(count + tabwidth, tabwidth) * tabwidth
360 else
361 return count, false
362 end
363 end

367 """
368 Removes leading indentation from string
369
370 Returns:
371
372 * `String` of multiline string, with leading indentation of `indent` removed
373 """
374 function unindent(str::AbstractString, indent::Int; tabwidth=8)
375 indent == 0 && return str
376 pos = start(str)
377 endpos = endof(str)
378 # Note: this loses the type of the original string
379 buf = IOBuffer(StringVector(endpos), true, true)
380 truncate(buf,0)
381 cutting = true
382 col = 0 # current column (0 based)
383 while pos <= endpos
384 ch, pos = next(str,pos)
385 if cutting
386 if ch == ' '
387 col += 1
388 elseif ch == '\t'
389 col = div(col + tabwidth, tabwidth) * tabwidth
390 elseif ch == '\n'
391 # Now we need to output enough indentation
392 for i = 1:col-indent
393 write(buf, ' ')
394 end
395 col = 0
396 write(buf, '\n')
397 else
398 cutting = false
399 # Now we need to output enough indentation to get to
400 # correct place
401 for i = 1:col-indent
402 write(buf, ' ')
403 end
404 col += 1
405 write(buf, ch)
406 end
407 elseif ch == '\t' # Handle internal tabs
408 upd = div(col + tabwidth, tabwidth) * tabwidth
409 # output the number of spaces that would have been seen
410 # with original indentation
411 for i = 1:(upd-col)
412 write(buf, ' ')
413 end
414 col = upd
415 elseif ch == '\n'
416 cutting = true
417 col = 0
418 write(buf, '\n')
419 else
420 col += 1
421 write(buf, ch)
422 end
423 end
424 # If we were still "cutting" when we hit the end of the string,
425 # we need to output the right number of spaces for the indentation
426 if cutting
427 for i = 1:col-indent
428 write(buf, ' ')
429 end
430 end
431 String(take!(buf))
432 end
