62 const StatusUninit = 0 # handle is allocated, but not initialized
63 const StatusInit = 1 # handle is valid, but not connected/active
64 const StatusConnecting = 2 # handle is in process of connecting
65 const StatusOpen = 3 # handle is usable
66 const StatusActive = 4 # handle is listening for read/write/connect events
67 const StatusClosing = 5 # handle is closing / being closed
68 const StatusClosed = 6 # handle is closed
69 const StatusEOF = 7 # handle is a TTY that has seen an EOF event
70 const StatusPaused = 8 # handle is Active, but not consuming events, and will transition to Open if it receives an event
71 function uv_status_string(x)
72 s = x.status
73 if x.handle == C_NULL
74 if s == StatusClosed
75 return "closed"
76 elseif s == StatusUninit
77 return "null"
78 end
79 return "invalid status"
80 elseif s == StatusUninit
81 return "uninit"
82 elseif s == StatusInit
83 return "init"
84 elseif s == StatusConnecting
85 return "connecting"
86 elseif s == StatusOpen
87 return "open"
88 elseif s == StatusActive
89 return "active"
90 elseif s == StatusPaused
91 return "paused"
92 elseif s == StatusClosing
93 return "closing"
94 elseif s == StatusClosed
95 return "closed"
96 elseif s == StatusEOF
97 return "eof"
98 end
99 return "invalid status"
100 end

263 function wait_readbyte(x::LibuvStream, c::UInt8)
264 if isopen(x) # fast path
265 search(x.buffer, c) > 0 && return
266 else
267 return
268 end
269 preserve_handle(x)
270 try
271 while isopen(x) && search(x.buffer, c) <= 0
272 start_reading(x) # ensure we are reading
273 wait(x.readnotify)
274 end
275 finally
276 if isempty(x.readnotify.waitq)
277 stop_reading(x) # stop reading iff there are currently no other read clients of the stream
278 end
279 unpreserve_handle(x)
280 end
281 nothing
282 end

284 function wait_readnb(x::LibuvStream, nb::Int)
285 if isopen(x) # fast path
286 nb_available(x.buffer) >= nb && return
287 else
288 return
289 end
290 oldthrottle = x.throttle
291 preserve_handle(x)
292 try
293 while isopen(x) && nb_available(x.buffer) < nb
294 x.throttle = max(nb, x.throttle)
295 start_reading(x) # ensure we are reading
296 wait(x.readnotify)
297 end
298 finally
299 if isempty(x.readnotify.waitq)
300 stop_reading(x) # stop reading iff there are currently no other read clients of the stream
301 end
302 if oldthrottle <= x.throttle <= nb
303 x.throttle = oldthrottle
304 end
305 unpreserve_handle(x)
306 end
307 nothing
308 end

463 function uv_readcb(handle::Ptr{Void}, nread::Cssize_t, buf::Ptr{Void})
464 stream_unknown_type = @handle_as handle LibuvStream
465 nrequested = ccall(:jl_uv_buf_len, Csize_t, (Ptr{Void},), buf)
466 function readcb_specialized(stream::LibuvStream, nread::Int, nrequested::UInt)
467 if nread < 0
468 if nread == UV_ENOBUFS && nrequested == 0
469 # remind the client that stream.buffer is full
470 notify(stream.readnotify)
471 elseif nread == UV_EOF
472 if isa(stream, TTY)
473 stream.status = StatusEOF # libuv called uv_stop_reading already
474 notify(stream.readnotify)
475 notify(stream.closenotify)
476 elseif stream.status != StatusClosing
477 # begin shutdown of the stream
478 ccall(:jl_close_uv, Void, (Ptr{Void},), stream.handle)
479 stream.status = StatusClosing
480 end
481 else
482 # This is a fatal connection error. Shutdown requests as per the usual
483 # close function won't work and libuv will fail with an assertion failure
484 ccall(:jl_forceclose_uv, Void, (Ptr{Void},), stream)
485 notify_error(stream.readnotify, UVError("read", nread))
486 end
487 else
488 notify_filled(stream.buffer, nread)
489 notify(stream.readnotify)
490 end
491
492 # Stop background reading when
493 # 1) there's nobody paying attention to the data we are reading
494 # 2) we have accumulated a lot of unread data OR
495 # 3) we have an alternate buffer that has reached its limit.
496 if stream.status == StatusPaused ||
497 (stream.status == StatusActive &&
498 ((nb_available(stream.buffer) >= stream.throttle) ||
499 (nb_available(stream.buffer) >= stream.buffer.maxsize)))
500 # save cycles by stopping kernel notifications from arriving
501 ccall(:uv_read_stop, Cint, (Ptr{Void},), stream)
502 stream.status = StatusOpen
503 end
504 nothing
505 end
506 readcb_specialized(stream_unknown_type, Int(nread), UInt(nrequested))
507 end

658 function start_reading(stream::LibuvStream)
659 if stream.status == StatusOpen
660 if !isreadable(stream)
661 error("tried to read a stream that is not readable")

664 # for a TTY on Windows, so ensure the status is set first

667 stream, uv_jl_alloc_buf::Ptr{Void}, uv_jl_readcb::Ptr{Void})
668 return ret
669 elseif stream.status == StatusPaused

671 return Int32(0)
672 elseif stream.status == StatusActive
673 return Int32(0)
674 else
675 return Int32(-1)
676 end

704 function readbytes!(s::LibuvStream, a::Vector{UInt8}, nb::Int)
705 sbuf = s.buffer
706 @assert sbuf.seekable == false
707 @assert sbuf.maxsize >= nb
708
709 if nb_available(sbuf) >= nb
710 return readbytes!(sbuf, a, nb)
711 end
712
713 if nb <= SZ_UNBUFFERED_IO # Under this limit we are OK with copying the array from the stream's buffer
714 wait_readnb(s, nb)
715 return readbytes!(sbuf, a, nb)
716 else
