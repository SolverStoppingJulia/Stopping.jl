34 """
35 init(; n::Integer, delay::Float64)
36
37 Configure the `delay` between backtraces (measured in seconds), and the number `n` of
38 instruction pointers that may be stored. Each instruction pointer corresponds to a single
39 line of code; backtraces generally consist of a long list of instruction pointers. Default
40 settings can be obtained by calling this function with no arguments, and each can be set
41 independently using keywords or in the order `(n, delay)`.
42 """
43 function init(; n::Union{Void,Integer} = nothing, delay::Union{Void,Float64} = nothing)
44 n_cur = ccall(:jl_profile_maxlen_data, Csize_t, ())
45 delay_cur = ccall(:jl_profile_delay_nsec, UInt64, ())/10^9
46 if n === nothing && delay === nothing
47 return Int(n_cur), delay_cur
48 end
49 nnew = (n === nothing) ? n_cur : n
50 delaynew = (delay === nothing) ? delay_cur : delay
51 init(nnew, delaynew)
52 end

100 Prints profiling results to `io` (by default, `STDOUT`). If you do not
101 supply a `data` vector, the internal buffer of accumulated backtraces
102 will be used.
103
104 The keyword arguments can be any combination of:
105
106 - `format` -- Determines whether backtraces are printed with (default, `:tree`) or without (`:flat`)
107 indentation indicating tree structure.
108
109 - `C` -- If `true`, backtraces from C and Fortran code are shown (normally they are excluded).
110
111 - `combine` -- If `true` (default), instruction pointers are merged that correspond to the same line of code.
112
113 - `maxdepth` -- Limits the depth higher than `maxdepth` in the `:tree` format.
114
115 - `sortedby` -- Controls the order in `:flat` format. `:filefuncline` (default) sorts by the source

117

182 """
183 flatten(btdata, lidict) -> (newdata::Vector{UInt64}, newdict::LineInfoFlatDict)
184
185 Produces "flattened" backtrace data. Individual instruction pointers
186 sometimes correspond to a multi-frame backtrace due to inlining; in
187 such cases, this function inserts fake instruction pointers for the
188 inlined calls, and returns a dictionary that is a 1-to-1 mapping
189 between instruction pointers and a single StackFrame.
190 """
191 function flatten(data::Vector, lidict::LineInfoDict)
192 # Makes fake instruction pointers, counting down from typemax(UInt)
193 newip = typemax(UInt64) - 1
194 taken = Set(keys(lidict)) # make sure we don't pick one that's already used
195 newdict = Dict{UInt64,StackFrame}()
196 newmap = Dict{UInt64,Vector{UInt64}}()
197 for (ip, trace) in lidict
198 if length(trace) == 1
199 newdict[ip] = trace[1]
200 else
201 newm = UInt64[]
202 for sf in trace
203 while newip ∈ taken && newip > 0
204 newip -= 1
205 end
206 newip == 0 && error("all possible instruction pointers used")
207 push!(newm, newip)
208 newdict[newip] = sf
209 newip -= 1
210 end
211 newmap[ip] = newm
212 end
213 end
214 newdata = UInt64[]
215 for ip in data
216 local ip::UInt64
217 if haskey(newmap, ip)
218 append!(newdata, newmap[ip])
219 else
220 push!(newdata, ip)
221 end
222 end
223 return (newdata, newdict)
224 end

292 """
293 fetch() -> data
294
295 Returns a reference to the internal buffer of backtraces. Note that subsequent operations,
296 like [`clear`](@ref), can affect `data` unless you first make a copy. Note that the
297 values in `data` have meaning only on this machine in the current session, because it
298 depends on the exact memory addresses used in JIT-compiling. This function is primarily for
299 internal use; [`retrieve`](@ref) may be a better choice for most users.
300 """
301 function fetch()
302 len = len_data()
303 maxlen = maxlen_data()
304 if (len == maxlen)
305 warn("""The profile data buffer is full; profiling probably terminated
306 before your program finished. To profile for longer runs, call Profile.init
307 with a larger buffer and/or larger delay.""")
308 end
309 return unsafe_wrap(Array, get_data_pointer(), (len,))
310 end

318 # Counts the number of times each line appears, at any nesting level
319 function count_flat(data::Vector{T}) where T<:Unsigned
320 linecount = Dict{T,Int}()
321 toskip = btskip
322 for ip in data
323 if toskip > 0
324 toskip -= 1
325 continue
326 end
327 if ip == 0
328 toskip = btskip
329 continue
330 end
331 linecount[ip] = get(linecount, ip, 0)+1
332 end
333 iplist = Vector{T}(0)
334 n = Vector{Int}(0)
335 for (k,v) in linecount
336 push!(iplist, k)
337 push!(n, v)
338 end
339 return (iplist, n)
340 end

342 function parse_flat(iplist, n, lidict::LineInfoFlatDict, C::Bool)
343 # Convert instruction pointers to names & line numbers
344 lilist = [lidict[ip] for ip in iplist]
345 # Keep only the interpretable ones
346 # The ones with no line number might appear multiple times in a single
347 # backtrace, giving the wrong impression about the total number of backtraces.
348 # Delete them too.
349 keep = .!Bool[x == UNKNOWN || x.line == 0 || (x.from_c && !C) for x in lilist]
350 n = n[keep]
351 lilist = lilist[keep]
352 return (lilist, n)

375 function print_flat(io::IO, lilist::Vector{StackFrame}, n::Vector{Int},
376 cols::Int, fmt::ProfileFormat)
377 p = liperm(lilist)
378 lilist = lilist[p]
379 n = n[p]
380 if fmt.combine
381 j = 1
382 for i = 2:length(lilist)
383 if lilist[i] == lilist[j]
384 n[j] += n[i]
385 n[i] = 0
386 else
387 j = i
388 end
389 end
390 keep = n .> 0
391 n = n[keep]
392 lilist = lilist[keep]
393 end
394 if fmt.sortedby == :count
395 p = sortperm(n)
396 n = n[p]
397 lilist = lilist[p]
398 end
399 wcounts = max(6, ndigits(maximum(n)))
400 maxline = 0
401 maxfile = 6
402 maxfunc = 10
403 for li in lilist
404 maxline = max(maxline, li.line)
405 maxfile = max(maxfile, length(string(li.file)))
406 maxfunc = max(maxfunc, length(string(li.func)))
407 end
408 wline = max(5, ndigits(maxline))
409 ntext = cols - wcounts - wline - 3
410 maxfunc += 25
411 if maxfile + maxfunc <= ntext
412 wfile = maxfile
413 wfunc = maxfunc
414 else
415 wfile = floor(Integer, 2*ntext/5)
416 wfunc = floor(Integer, 3*ntext/5)
417 end
418 println(io, lpad("Count", wcounts, " "), " ", rpad("File", wfile, " "), " ",
419 lpad("Line", wline, " "), " ", rpad("Function", wfunc, " "))
420 for i = 1:length(n)
421 n[i] < fmt.mincount && continue
422 li = lilist[i]
423 Base.print(io, lpad(string(n[i]), wcounts, " "), " ")
424 Base.print(io, rpad(rtruncto(string(li.file), wfile), wfile, " "), " ")
425 Base.print(io, lpad(string(li.line), wline, " "), " ")
426 fname = string(li.func)
427 if !li.from_c && !isnull(li.linfo)
428 fname = sprint(show_spec_linfo, li)
429 end
430 Base.print(io, rpad(ltruncto(fname, wfunc), wfunc, " "))
431 println(io)
432 end
433 nothing

436 ## A tree representation
437 # Identify and counts repetitions of all unique backtraces
438 function tree_aggregate(data::Vector{UInt64})
439 iz = find(iszero, data) # find the breaks between backtraces
440 treecount = Dict{Vector{UInt64},Int}()
441 istart = 1 + btskip
442 for iend in iz
443 tmp = data[iend - 1 : -1 : istart]
444 treecount[tmp] = get(treecount, tmp, 0) + 1
445 istart = iend + 1 + btskip
446 end
447 bt = Vector{Vector{UInt64}}(0)
448 counts = Vector{Int}(0)
449 for (k, v) in treecount
450 if !isempty(k)
451 push!(bt, k)
452 push!(counts, v)
453 end
454 end
455 return (bt, counts)
456 end

460 function tree_format(lilist::Vector{StackFrame}, counts::Vector{Int}, level::Int, cols::Int)
461 nindent = min(cols>>1, level)
462 ndigcounts = ndigits(maximum(counts))
463 ndigline = maximum([tree_format_linewidth(x) for x in lilist])
464 ntext = cols - nindent - ndigcounts - ndigline - 5
465 widthfile = floor(Integer, 0.4ntext)
466 widthfunc = floor(Integer, 0.6ntext)
467 strs = Vector{String}(length(lilist))
468 showextra = false
469 if level > nindent
470 nextra = level - nindent
471 nindent -= ndigits(nextra) + 2
472 showextra = true
473 end
474 for i = 1:length(lilist)
475 li = lilist[i]
476 if li != UNKNOWN
477 base = " "^nindent
478 if showextra
479 base = string(base, "+", nextra, " ")
480 end
481 if li.line == li.pointer
482 strs[i] = string(base,
483 rpad(string(counts[i]), ndigcounts, " "),
484 " ",
485 "unknown function (pointer: 0x",
486 hex(li.pointer,2*sizeof(Ptr{Void})),
487 ")")
488 else
489 fname = string(li.func)
490 if !li.from_c && !isnull(li.linfo)
491 fname = sprint(show_spec_linfo, li)
492 end
493 strs[i] = string(base,
494 rpad(string(counts[i]), ndigcounts, " "),
495 " ",
496 rtruncto(string(li.file), widthfile),
497 ":",
498 li.line == -1 ? "?" : string(li.line),
499 "; ",
500 ltruncto(fname, widthfunc))
501 end
502 else
503 strs[i] = ""
504 end
505 end
506 return strs
507 end

509 # Print a "branch" starting at a particular level. This gets called recursively.
510 function tree(io::IO, bt::Vector{Vector{UInt64}}, counts::Vector{Int},
511 lidict::LineInfoFlatDict, level::Int, cols::Int, fmt::ProfileFormat, noisefloor::Int)
512 if level > fmt.maxdepth
513 return
514 end
515 # Organize backtraces into groups that are identical up to this level
516 if fmt.combine
517 # Combine based on the line information
518 d = Dict{StackFrame,Vector{Int}}()
519 for i = 1:length(bt)
520 ip = bt[i][level + 1]
521 key = lidict[ip]
522 indx = Base.ht_keyindex(d, key)
523 if haskey(d, key)
524 push!(d[key], i)
525 else
526 d[key] = [i]
527 end
528 end
529 # Generate counts
530 dlen = length(d)
531 lilist = Vector{StackFrame}(dlen)
532 group = Vector{Vector{Int}}(dlen)
533 n = Vector{Int}(dlen)
534 i = 1
535 for (key, v) in d
536 lilist[i] = key
537 group[i] = v
538 n[i] = sum(counts[v])
539 i += 1
540 end
541 else
542 # Combine based on the instruction pointer
543 d = Dict{UInt64,Vector{Int}}()
544 for i = 1:length(bt)
545 key = bt[i][level+1]
546 if haskey(d, key)
547 push!(d[key], i)
548 else
549 d[key] = [i]
550 end
551 end
552 # Generate counts, and do the code lookup
553 dlen = length(d)
554 lilist = Vector{StackFrame}(dlen)
555 group = Vector{Vector{Int}}(dlen)
556 n = Vector{Int}(dlen)
557 i = 1
558 for (key, v) in d
559 lilist[i] = lidict[key]
560 group[i] = v
561 n[i] = sum(counts[v])
562 i += 1
563 end
564 end
565 # Order the line information
566 if length(lilist) > 1
567 p = liperm(lilist)
568 lilist = lilist[p]
569 group = group[p]
570 n = n[p]
571 end
572 # Generate the string for each line
573 strs = tree_format(lilist, n, level, cols)
574 # Recurse to the next level
575 len = Int[length(x) for x in bt]
576 for i = 1:length(lilist)
577 n[i] < fmt.mincount && continue
578 n[i] < noisefloor && continue
579 if !isempty(strs[i])
580 println(io, strs[i])
581 end
582 idx = group[i]
583 keep = len[idx] .> level+1
584 if any(keep)
585 idx = idx[keep]
586 tree(io, bt[idx], counts[idx], lidict, level + 1, cols, fmt, fmt.noisefloor > 0 ? floor(Int, fmt.noisefloor * sqrt(n[i])) : 0)
587 end
588 end
589 nothing
590 end

614 function callersf(matchfunc::Function, bt::Vector, lidict::LineInfoDict)
615 counts = Dict{StackFrame, Int}()
616 lastmatched = false
617 for id in bt
618 if id == 0
619 lastmatched = false
620 continue
621 end
622 li = lidict[id]
623 if lastmatched
624 if haskey(counts, li)
625 counts[li] += 1
626 else
627 counts[li] = 1
628 end
629 end
630 lastmatched = matchfunc(li)
631 end
632 k = collect(keys(counts))
633 v = collect(values(counts))
634 p = sortperm(v, rev=true)
635 return [(v[i], k[i]) for i in p]
636 end

657 # Order alphabetically (file, function) and then by line number
658 function liperm(lilist::Vector{StackFrame})
659 comb = Vector{String}(length(lilist))
660 for i = 1:length(lilist)
661 li = lilist[i]
662 if li != UNKNOWN
663 comb[i] = @sprintf("%s:%s:%06d", li.file, li.func, li.line)

666 end
667 end
