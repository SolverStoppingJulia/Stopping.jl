97 # to make StepRange constructor inlineable, so optimizer can see `step` value
98 function steprange_last(start::T, step, stop) where T
99 if isa(start,AbstractFloat) || isa(step,AbstractFloat)
100 throw(ArgumentError("StepRange should not be used with floating point"))
101 end
102 z = zero(step)
103 step == z && throw(ArgumentError("step cannot be zero"))
104
105 if stop == start
106 last = stop
107 else
108 if (step > z) != (stop > start)
109 last = steprange_last_empty(start, step, stop)
110 else
111 diff = stop - start
112 if T<:Signed && (diff > zero(diff)) != (stop > start)
113 # handle overflowed subtraction with unsigned rem
114 if diff > zero(diff)
115 remain = -convert(T, unsigned(-diff) % step)
116 else
117 remain = convert(T, unsigned(diff) % step)
118 end
119 else
120 remain = steprem(start,stop,step)
121 end
122 last = stop - remain
123 end
124 end
125 last
126 end

128 function steprange_last_empty(start::Integer, step, stop)
129 # empty range has a special representation where stop = start-1
130 # this is needed to avoid the wrap-around that can happen computing
131 # start - step, which leads to a range that looks very large instead
132 # of empty.
133 if step > zero(step)
134 last = start - oneunit(stop-start)
135 else
136 last = start + oneunit(stop-start)
137 end
138 last
139 end

259 """
260 `print_range(io, r)` prints out a nice looking range r in terms of its elements
261 as if it were `collect(r)`, dependent on the size of the
262 terminal, and taking into account whether compact numbers should be shown.
263 It figures out the width in characters of each element, and if they
264 end up too wide, it shows the first and last elements separated by a
265 horizontal elipsis. Typical output will look like `1.0,2.0,3.0,â¦,4.0,5.0,6.0`.
266
267 `print_range(io, r, pre, sep, post, hdots)` uses optional
268 parameters `pre` and `post` characters for each printed row,
269 `sep` separator string between printed elements,
270 `hdots` string for the horizontal ellipsis.
271 """
272 function print_range(io::IO, r::Range,
273 pre::AbstractString = " ",
274 sep::AbstractString = ",",
275 post::AbstractString = "",
276 hdots::AbstractString = ",\u2026,") # horiz ellipsis
277 # This function borrows from print_matrix() in show.jl
278 # and should be called by show and display
279 limit = get(io, :limit, false)
280 sz = displaysize(io)
281 if !haskey(io, :compact)
282 io = IOContext(io, :compact => true)
283 end
284 screenheight, screenwidth = sz[1] - 4, sz[2]
285 screenwidth -= length(pre) + length(post)
286 postsp = ""
287 sepsize = length(sep)
288 m = 1 # treat the range as a one-row matrix
289 n = length(r)
290 # Figure out spacing alignments for r, but only need to examine the
291 # left and right edge columns, as many as could conceivably fit on the
292 # screen, with the middle columns summarized by horz, vert, or diag ellipsis
293 maxpossiblecols = div(screenwidth, 1+sepsize) # assume each element is at least 1 char + 1 separator
294 colsr = n <= maxpossiblecols ? (1:n) : [1:div(maxpossiblecols,2)+1; (n-div(maxpossiblecols,2)):n]
295 rowmatrix = reshape(r[colsr], 1, length(colsr)) # treat the range as a one-row matrix for print_matrix_row
296 A = alignment(io, rowmatrix, 1:m, 1:length(rowmatrix), screenwidth, screenwidth, sepsize) # how much space range takes
297 if n <= length(A) # cols fit screen, so print out all elements
298 print(io, pre) # put in pre chars
299 print_matrix_row(io,rowmatrix,A,1,1:n,sep) # the entire range
300 print(io, post) # add the post characters
301 else # cols don't fit so put horiz ellipsis in the middle
302 # how many chars left after dividing width of screen in half
303 # and accounting for the horiz ellipsis
304 c = div(screenwidth-length(hdots)+1,2)+1 # chars remaining for each side of rowmatrix
305 alignR = reverse(alignment(io, rowmatrix, 1:m, length(rowmatrix):-1:1, c, c, sepsize)) # which cols of rowmatrix to put on the right
306 c = screenwidth - sum(map(sum,alignR)) - (length(alignR)-1)*sepsize - length(hdots)
307 alignL = alignment(io, rowmatrix, 1:m, 1:length(rowmatrix), c, c, sepsize) # which cols of rowmatrix to put on the left
308 print(io, pre) # put in pre chars
309 print_matrix_row(io, rowmatrix,alignL,1,1:length(alignL),sep) # left part of range
310 print(io, hdots) # horizontal ellipsis
311 print_matrix_row(io, rowmatrix,alignR,1,length(rowmatrix)-length(alignR)+1:length(rowmatrix),sep) # right part of range
312 print(io, post) # post chars
313 end
314 end

577 function ==(r::Range, s::Range)
578 lr = length(r)
579 if lr != length(s)
580 return false
581 end
582 u, v = start(r), start(s)
583 while !done(r, u)
584 x, u = next(r, u)
585 y, v = next(s, v)
586 if x != y
587 return false
588 end
589 end
590 return true
591 end

603 function intersect(r::AbstractUnitRange{<:Integer}, s::StepRange{<:Integer})
604 if isempty(s)
605 range(first(r), 0)
606 elseif step(s) == 0
607 intersect(first(s), r)
608 elseif step(s) < 0
609 intersect(r, reverse(s))
610 else
611 sta = first(s)
612 ste = step(s)
613 sto = last(s)
614 lo = first(r)
615 hi = last(r)
616 i0 = max(sta, lo + mod(sta - lo, ste))
617 i1 = min(sto, hi - mod(hi - sta, ste))
618 i0:ste:i1
619 end
620 end

630 function intersect(r::StepRange, s::StepRange)
631 if isempty(r) || isempty(s)
632 return range(first(r), step(r), 0)
633 elseif step(s) < 0
634 return intersect(r, reverse(s))
635 elseif step(r) < 0
636 return reverse(intersect(reverse(r), s))
637 end
638
639 start1 = first(r)
640 step1 = step(r)
641 stop1 = last(r)
642 start2 = first(s)
643 step2 = step(s)
644 stop2 = last(s)
645 a = lcm(step1, step2)
646
647 # if a == 0
648 # # One or both ranges have step 0.
649 # if step1 == 0 && step2 == 0
650 # return start1 == start2 ? r : Range(start1, 0, 0)
651 # elseif step1 == 0
652 # return start2 <= start1 <= stop2 && rem(start1 - start2, step2) == 0 ? r : Range(start1, 0, 0)
653 # else
654 # return start1 <= start2 <= stop1 && rem(start2 - start1, step1) == 0 ? (start2:step1:start2) : Range(start1, step1, 0)
655 # end
656 # end
657
658 g, x, y = gcdx(step1, step2)
659
660 if rem(start1 - start2, g) != 0
661 # Unaligned, no overlap possible.
662 return range(start1, a, 0)
663 end
664
665 z = div(start1 - start2, g)
666 b = start1 - x * z * step1
667 # Possible points of the intersection of r and s are
668 # ..., b-2a, b-a, b, b+a, b+2a, ...
669 # Determine where in the sequence to start and stop.
670 m = max(start1 + mod(b - start1, a), start2 + mod(b - start2, a))
671 n = min(stop1 - mod(stop1 - b, a), stop2 - mod(stop2 - b, a))
672 m:a:n
673 end

684 function _findin(r::Range{<:Integer}, span::AbstractUnitRange{<:Integer})
685 local ifirst
686 local ilast
687 fspan = first(span)
688 lspan = last(span)
689 fr = first(r)
690 lr = last(r)
691 sr = step(r)
692 if sr > 0
693 ifirst = fr >= fspan ? 1 : ceil(Integer,(fspan-fr)/sr)+1
694 ilast = lr <= lspan ? length(r) : length(r) - ceil(Integer,(lr-lspan)/sr)
695 elseif sr < 0
696 ifirst = fr <= lspan ? 1 : ceil(Integer,(lspan-fr)/sr)+1
697 ilast = lr >= fspan ? length(r) : length(r) - ceil(Integer,(lr-fspan)/sr)
698 else
699 ifirst = fr >= fspan ? 1 : length(r)+1
700 ilast = fr <= lspan ? length(r) : 0
701 end
702 ifirst, ilast
703 end

879 function _define_range_op(f::ANY)
880 @eval begin
881 function $f(r1::OrdinalRange, r2::OrdinalRange)
882 r1l = length(r1)
883 (r1l == length(r2) ||
884 throw(DimensionMismatch("argument dimensions must match")))
885 range($f(first(r1),first(r2)), $f(step(r1),step(r2)), r1l)
886 end
887
888 function $f(r1::LinSpace{T}, r2::LinSpace{T}) where T
889 len = r1.len
890 (len == r2.len ||
891 throw(DimensionMismatch("argument dimensions must match")))
892 linspace(convert(T, $f(first(r1), first(r2))),
893 convert(T, $f(last(r1), last(r2))), len)
894 end
895
896 $f(r1::Union{StepRangeLen, OrdinalRange, LinSpace},
897 r2::Union{StepRangeLen, OrdinalRange, LinSpace}) =
898 $f(promote_noncircular(r1, r2)...)
899 end
900 end
