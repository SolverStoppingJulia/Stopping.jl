63 """
64 keys(a::Associative)
65
66 Return an iterator over all keys in a collection.
67 `collect(keys(a))` returns an array of keys.
68 Since the keys are stored internally in a hash table,
69 the order in which they are returned may vary.
70 But `keys(a)` and `values(a)` both iterate `a` and
71 return the elements in the same order.
72
73 ```jldoctest
74 julia> a = Dict('a'=>2, 'b'=>3)
75 Dict{Char,Int64} with 2 entries:
76 'b' => 3
77 'a' => 2
78
79 julia> collect(keys(a))
80 2-element Array{Char,1}:
81 'b'
82 'a'
83 ```
84 """

149 """
150 merge!(combine, d::Associative, others::Associative...)
151
152 Update collection with pairs from the other collections.
153 Values with the same key will be combined using the
154 combiner function.
155
156 ```jldoctest
157 julia> d1 = Dict(1 => 2, 3 => 4);
158
159 julia> d2 = Dict(1 => 4, 4 => 5);
160
161 julia> merge!(+, d1, d2);
162
163 julia> d1
164 Dict{Int64,Int64} with 3 entries:
165 4 => 5
166 3 => 4
167 1 => 6
168
169 julia> merge!(-, d1, d1);
170
171 julia> d1
172 Dict{Int64,Int64} with 3 entries:
173 4 => 0
174 3 => 0
175 1 => 0
176 ```
177 """
178 function merge!(combine::Function, d::Associative, others::Associative...)
179 for other in others
180 for (k,v) in other
181 d[k] = haskey(d, k) ? combine(d[k], v) : v
182 end
183 end
184 return d
185 end

298 function filter!(f, d::Associative)
299 badkeys = Vector{keytype(d)}(0)
300 for (k,v) in d
301 # don't delete!(d, k) here, since associative types
302 # may not support mutation during iteration
303 f(k,v) || push!(badkeys, k)
304 end
305 for k in badkeys
306 delete!(d, k)
307 end
308 return d
309 end
