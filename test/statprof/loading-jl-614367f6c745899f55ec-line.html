16 elseif is_apple()
17 # HFS+ filesystem is case-preserving. The getattrlist API returns
18 # a case-preserved filename. In the rare event that HFS+ is operating
19 # in case-sensitive mode, this will still work but will be redundant.
20
21 # Constants from <sys/attr.h>
22 const ATRATTR_BIT_MAP_COUNT = 5
23 const ATTR_CMN_NAME = 1
24 const BITMAPCOUNT = 1
25 const COMMONATTR = 5
26 const FSOPT_NOFOLLOW = 1 # Don't follow symbolic links
27
28 const attr_list = zeros(UInt8, 24)
29 attr_list[BITMAPCOUNT] = ATRATTR_BIT_MAP_COUNT
30 attr_list[COMMONATTR] = ATTR_CMN_NAME

32 # This essentially corresponds to the following C code:
33 # attrlist attr_list;
34 # memset(&attr_list, 0, sizeof(attr_list));
35 # attr_list.bitmapcount = ATTR_BIT_MAP_COUNT;
36 # attr_list.commonattr = ATTR_CMN_NAME;
37 # struct Buffer {
38 # u_int32_t total_length;
39 # u_int32_t filename_offset;
40 # u_int32_t filename_length;
41 # char filename[max_filename_length];
42 # };
43 # Buffer buf;
44 # getattrpath(path, &attr_list, &buf, sizeof(buf), FSOPT_NOFOLLOW);
45 function isfile_casesensitive(path)
46 isfile(path) || return false
47 path_basename = String(basename(path))
48 local casepreserved_basename
49 const header_size = 12
50 buf = Vector{UInt8}(length(path_basename) + header_size + 1)
51 while true
52 ret = ccall(:getattrlist, Cint,
53 (Cstring, Ptr{Void}, Ptr{Void}, Csize_t, Culong),
54 path, attr_list, buf, sizeof(buf), FSOPT_NOFOLLOW)
55 systemerror(:getattrlist, ret â  0)
56 filename_length = unsafe_load(
57 convert(Ptr{UInt32}, pointer(buf) + 8))
58 if (filename_length + header_size) > length(buf)
59 resize!(buf, filename_length + header_size)
60 continue
61 end
62 casepreserved_basename =
63 view(buf, (header_size+1):(header_size+filename_length-1))
64 break
65 end
66 # Hack to compensate for inability to create a string from a subarray with no allocations.
67 Vector{UInt8}(path_basename) == casepreserved_basename && return true
68
69 # If there is no match, it's possible that the file does exist but HFS+
70 # performed unicode normalization. See https://developer.apple.com/library/mac/qa/qa1235/_index.html.
71 isascii(path_basename) && return false
72 Vector{UInt8}(normalize_string(path_basename, :NFD)) == casepreserved_basename
73 end

324 Specify whether the file calling this function is precompilable. If `isprecompilable` is
325 `true`, then `__precompile__` throws an exception when the file is loaded by
326 `using`/`import`/`require` *unless* the file is being precompiled, and in a module file it
327 causes the module to be automatically precompiled when it is imported. Typically,

412 function _require(mod::Symbol)
413 # dependency-tracking is only used for one top-level include(path),
414 # and is not applied recursively to imported modules:
415 old_track_dependencies = _track_dependencies[]
416 _track_dependencies[] = false
417 DEBUG_LOADING[] = haskey(ENV, "JULIA_DEBUG_LOADING")
418
419 global toplevel_load
420 loading = get(package_locks, mod, false)
421 if loading !== false
422 # load already in progress for this module
423 wait(loading)
424 return
425 end
426 package_locks[mod] = Condition()
427
428 last = toplevel_load::Bool
429 try
430 toplevel_load = false
431 # perform the search operation to select the module file require intends to load
432 name = string(mod)
433 path = find_in_node_path(name, nothing, 1)
434 if path === nothing
435 throw(ArgumentError("Module $name not found in current path.\nRun `Pkg.add(\"$name\")` to install the $name package."))
436 end
437
438 # attempt to load the module file via the precompile cache locations
439 doneprecompile = false
440 if JLOptions().use_compilecache != 0
441 doneprecompile = _require_search_from_serialized(1, mod, path, last)
442 if !isa(doneprecompile, Bool)
443 return # success
444 end
445 end
446
447 # if the module being required was supposed to have a particular version
448 # but it was not handled by the precompile loader, complain
449 for (concrete_mod, concrete_uuid) in _concrete_dependencies
450 if mod === concrete_mod
451 warn("""Module $mod with uuid $concrete_uuid is missing from the cache.
452 This may mean module $mod does not support precompilation but is imported by a module that does.""")
453 if JLOptions().incremental != 0
454 # during incremental precompilation, this should be fail-fast
455 throw(PrecompilableError(false))
456 end
457 end
458 end
459
460 if doneprecompile === true || JLOptions().incremental != 0
461 # spawn off a new incremental pre-compile task from node 1 for recursive `require` calls
462 # or if the require search declared it was pre-compiled before (and therefore is expected to still be pre-compilable)
463 cachefile = compilecache(mod)
464 m = _require_from_serialized(1, mod, cachefile, last)
465 if isa(m, Exception)
466 warn("The call to compilecache failed to create a usable precompiled cache file for module $name. Got:")
467 warn(m, prefix="WARNING: ")
468 # fall-through, TODO: disable __precompile__(true) error so that the normal include will succeed
469 else
470 return # success
471 end
472 end
473
474 # just load the file normally via include
475 # for unknown dependencies
476 try
477 if last && myid() == 1 && nprocs() > 1
478 # include on node 1 first to check for PrecompilableErrors
479 eval(Main, :(Base.include_from_node1($path)))
480
481 # broadcast top-level import/using from node 1 (only)
482 @sync begin
483 for p in filter(x -> x != 1, procs())
484 @async remotecall_fetch(p) do
485 eval(Main, :(Base.include_from_node1($path); nothing))
486 end
487 end
488 end
489 else
490 eval(Main, :(Base.include_from_node1($path)))
491 end
492 catch ex
493 if doneprecompile === true || JLOptions().use_compilecache == 0 || !precompilableerror(ex, true)
494 rethrow() # rethrow non-precompilable=true errors
495 end
496 # the file requested `__precompile__`, so try to build a cache file and use that
497 cachefile = compilecache(mod)
498 m = _require_from_serialized(1, mod, cachefile, last)
499 if isa(m, Exception)
500 warn(m, prefix="WARNING: ")
501 # TODO: disable __precompile__(true) error and do normal include instead of error
502 error("Module $mod declares __precompile__(true) but require failed to create a usable precompiled cache file.")
503 end
504 end
505 finally
506 toplevel_load = last
507 loading = pop!(package_locks, mod)
508 notify(loading, all=true)
509 _track_dependencies[] = old_track_dependencies
510 end
511 nothing
512 end

671 This can be used to reduce package load times. Cache files are stored in
672 `LOAD_CACHE_PATH[1]`, which defaults to `~/.julia/lib/VERSION`. See
673 [Module initialization and precompilation](@ref)
674 for important notes.
675 """
676 function compilecache(name::String)
677 myid() == 1 || error("can only precompile from node 1")
678 # decide where to get the source file from
679 path = find_in_path(name, nothing)
680 path === nothing && throw(ArgumentError("$name not found in path"))
681 path = String(path)
682 # decide where to put the resulting cache file
683 cachepath = LOAD_CACHE_PATH[1]
684 if !isdir(cachepath)
685 mkpath(cachepath)
686 end
687 cachefile::String = abspath(cachepath, name*".ji")
688 # build up the list of modules that we want the precompile process to preserve
689 concrete_deps = copy(_concrete_dependencies)
690 for existing in names(Main)
691 if isdefined(Main, existing)
692 mod = getfield(Main, existing)
693 if isa(mod, Module) && !(mod === Main || mod === Core || mod === Base)
694 mod = mod::Module
695 if module_parent(mod) === Main && module_name(mod) === existing
696 push!(concrete_deps, (existing, module_uuid(mod)))
697 end
698 end
699 end
700 end
701 # run the expression and cache the result
702 if isinteractive() || DEBUG_LOADING[]
703 if isfile(cachefile)
704 info("Recompiling stale cache file $cachefile for module $name.")
705 else
706 info("Precompiling module $name.")
707 end
708 end
709 if !success(create_expr_cache(path, cachefile, concrete_deps))
710 error("Failed to precompile $name to $cachefile.")
711 end
712 return cachefile
713 end

715 module_uuid(m::Module) = ccall(:jl_module_uuid, UInt64, (Any,), m)

717 isvalid_cache_header(f::IOStream) = 0 != ccall(:jl_read_verify_header, Cint, (Ptr{Void},), f.ios)

719 function parse_cache_header(f::IO)
720 modules = Dict{Symbol,UInt64}()
721 while true
722 n = ntoh(read(f, Int32))
723 n == 0 && break
724 sym = Symbol(read(f, n)) # module symbol
725 uuid = ntoh(read(f, UInt64)) # module UUID (mostly just a timestamp)
726 modules[sym] = uuid
727 end
728 totbytes = ntoh(read(f, Int64)) # total bytes for file dependencies
729 # read the list of files
730 files = Tuple{String,Float64}[]
731 while true
732 n = ntoh(read(f, Int32))
733 n == 0 && break
734 totbytes -= 4 + n + 8
735 @assert n >= 0 "EOF while reading cache header" # probably means this wasn't a valid file to be read by Base.parse_cache_header
736 push!(files, (String(read(f, n)), ntoh(read(f, Float64))))
737 end
738 @assert totbytes == 4 "header of cache file appears to be corrupt"
739 # read the list of modules that are required to be present during loading
740 required_modules = Dict{Symbol,UInt64}()
741 while true
742 n = ntoh(read(f, Int32))
743 n == 0 && break
744 sym = Symbol(read(f, n)) # module symbol
745 uuid = ntoh(read(f, UInt64)) # module UUID
746 required_modules[sym] = uuid
747 end
748 return modules, files, required_modules
749 end

751 function parse_cache_header(cachefile::String)
752 io = open(cachefile, "r")
753 try
754 !isvalid_cache_header(io) && throw(ArgumentError("Invalid header in cache file $cachefile."))
755 return parse_cache_header(io)
756 finally
757 close(io)
758 end
759 end

761 function cache_dependencies(f::IO)
762 defs, files, modules = parse_cache_header(f)

776 function stale_cachefile(modpath::String, cachefile::String)
777 io = open(cachefile, "r")
778 try
779 if !isvalid_cache_header(io)
780 DEBUG_LOADING[] && info("JL_DEBUG_LOADING: Rejecting cache file $cachefile due to it containing an invalid cache header.")
781 return true # invalid cache file
782 end
783 modules, files, required_modules = parse_cache_header(io)
784
785 # Check if transitive dependencies can be fullfilled
786 for mod in keys(required_modules)
787 if mod == :Main || mod == :Core || mod == :Base
788 continue
789 # Module is already loaded
790 elseif isbindingresolved(Main, mod)
791 continue
792 end
793 name = string(mod)
794 path = find_in_node_path(name, nothing, 1)
795 if path === nothing
796 return true # Won't be able to fullfill dependency
797 end
798 end
799
800 # check if this file is going to provide one of our concrete dependencies
801 # or if it provides a version that conflicts with our concrete dependencies
802 # or neither
803 for (mod, uuid_req) in _concrete_dependencies
804 uuid = get(modules, mod, UInt64(0))
805 if uuid !== UInt64(0)
806 if uuid === uuid_req
807 return false # this is the file we want
808 end
809 DEBUG_LOADING[] && info("JL_DEBUG_LOADING: Rejecting cache file $cachefile because it provides the wrong uuid (got $uuid) for $mod (want $uuid_req).")
810 return true # cachefile doesn't provide the required version of the dependency
811 end
812 end
813
814 # now check if this file is fresh relative to its source files
815 if !samefile(files[1][1], modpath)
816 DEBUG_LOADING[] && info("JL_DEBUG_LOADING: Rejecting cache file $cachefile because it is for file $(files[1][1])) not file $modpath.")
817 return true # cache file was compiled from a different path
818 end
819 for (f, ftime_req) in files
820 # Issue #13606: compensate for Docker images rounding mtimes
821 # Issue #20837: compensate for GlusterFS truncating mtimes to microseconds
822 ftime = mtime(f)
823 if ftime != ftime_req && ftime != floor(ftime_req) && ftime != trunc(ftime_req, 6)
824 DEBUG_LOADING[] && info("JL_DEBUG_LOADING: Rejecting stale cache file $cachefile (mtime $ftime_req) because file $f (mtime $ftime) has changed.")
825 return true
826 end
827 end
828 return false # fresh cachefile
829 finally
830 close(io)
831 end
832 end
