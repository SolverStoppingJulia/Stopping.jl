36 function mapfoldl_impl(f, op, v0, itr, i)
37 # Unroll the while loop once; if v0 is known, the call to op may
38 # be evaluated at compile time
39 if done(itr, i)
40 return r_promote(op, v0)
41 else
42 (x, i) = next(itr, i)
43 v = op(r_promote(op, v0), f(x))
44 while !done(itr, i)
45 @inbounds (x, i) = next(itr, i)
46 v = op(v, f(x))
47 end
48 return v
49 end

104 function mapfoldr_impl(f, op, v0, itr, i::Integer)
105 # Unroll the while loop once; if v0 is known, the call to op may
106 # be evaluated at compile time
107 if isempty(itr) || i == 0
108 return r_promote(op, v0)
109 else
110 x = itr[i]
111 v = op(f(x), r_promote(op, v0))
112 while i > 1
113 x = itr[i -= 1]
114 v = op(f(x), v)
115 end
116 return v
117 end
118 end

168 ## reduce & mapreduce

170 # `mapreduce_impl()` is called by `mapreduce()` (via `_mapreduce()`, when `A`
171 # supports linear indexing) and does actual calculations (for `A[ifirst:ilast]` subset).
172 # For efficiency, no parameter validity checks are done, it's the caller's responsibility.
173 # `ifirst:ilast` range is assumed to be a valid non-empty subset of `A` indices.
174
175 # This is a generic implementation of `mapreduce_impl()`,
176 # certain `op` (e.g. `min` and `max`) may have their own specialized versions.
177 function mapreduce_impl(f, op, A::AbstractArray, ifirst::Integer, ilast::Integer, blksize::Int=pairwise_blocksize(f, op))
178 if ifirst == ilast
179 @inbounds a1 = A[ifirst]
180 return r_promote(op, f(a1))
181 elseif ifirst + blksize > ilast
182 # sequential portion
183 @inbounds a1 = A[ifirst]
184 @inbounds a2 = A[ifirst+1]
185 v = op(r_promote(op, f(a1)), r_promote(op, f(a2)))
186 @simd for i = ifirst + 2 : ilast
187 @inbounds ai = A[i]
188 v = op(v, f(ai))
189 end
190 return v
191 else
192 # pairwise portion
193 imid = (ifirst + ilast) >> 1
194 v1 = mapreduce_impl(f, op, A, ifirst, imid, blksize)
195 v2 = mapreduce_impl(f, op, A, imid+1, ilast, blksize)
196 return op(v1, v2)
197 end
198 end

208 """
209 mapreduce(f, op, v0, itr)
210
211 Apply function `f` to each element in `itr`, and then reduce the result using the binary
212 function `op`. `v0` must be a neutral element for `op` that will be returned for empty
213 collections. It is unspecified whether `v0` is used for non-empty collections.
214
215 [`mapreduce`](@ref) is functionally equivalent to calling `reduce(op, v0,
216 map(f, itr))`, but will in general execute faster since no intermediate collection needs to
217 be created. See documentation for [`reduce`](@ref) and [`map`](@ref).
218
219 ```jldoctest
220 julia> mapreduce(x->x^2, +, [1:3;]) # == 1 + 4 + 9
221 14
222 ```
223
224 The associativity of the reduction is implementation-dependent. Additionally, some
225 implementations may reuse the return value of `f` for elements that appear multiple times in
226 `itr`. Use [`mapfoldl`](@ref) or [`mapfoldr`](@ref) instead for
227 guaranteed left or right associativity and invocation of `f` for every value.
228 """
229 mapreduce(f, op, v0, itr) = mapfoldl(f, op, v0, itr)

289 """
290 reduce(op, v0, itr)
291
292 Reduce the given collection `√É¬¨tr` with the given binary operator `op`. `v0` must be a
293 neutral element for `op` that will be returned for empty collections. It is unspecified
294 whether `v0` is used for non-empty collections.
295
296 Reductions for certain commonly-used operators have special implementations which should be
297 used instead: `maximum(itr)`, `minimum(itr)`, `sum(itr)`, `prod(itr)`, `any(itr)`,
298 `all(itr)`.
299
300 The associativity of the reduction is implementation dependent. This means that you can't
301 use non-associative operations like `-` because it is undefined whether `reduce(-,[1,2,3])`
302 should be evaluated as `(1-2)-3` or `1-(2-3)`. Use [`foldl`](@ref) or
303 [`foldr`](@ref) instead for guaranteed left or right associativity.
304
305 Some operations accumulate error, and parallelism will also be easier if the reduction can
306 be executed in groups. Future versions of Julia might change the algorithm. Note that the
307 elements are not reordered if you use an ordered collection.
308
309 # Examples
310
311 ```jldoctest
312 julia> reduce(*, 1, [2; 3; 4])
313 24
314 ```
315 """
316 reduce(op, v0, itr) = mapreduce(identity, op, v0, itr)

366 """
367 sum_kbn(A)
368
369 Returns the sum of all elements of `A`, using the Kahan-Babuska-Neumaier compensated
370 summation algorithm for additional accuracy.
371 """
372 function sum_kbn(A)
373 T = _default_eltype(typeof(A))
374 c = r_promote(+, zero(T)::T)
375 i = start(A)
376 if done(A, i)
377 return c
378 end
379 Ai, i = next(A, i)
380 s = Ai - c
381 while !(done(A, i))
382 Ai, i = next(A, i)
383 t = s + Ai
384 if abs(s) >= abs(Ai)
385 c -= ((s-t) + Ai)
386 else
387 c -= ((Ai-t) + s)
388 end
389 s = t
390 end
391 s - c
392 end

476 """
477 extrema(itr) -> Tuple
478
479 Compute both the minimum and maximum element in a single pass, and return them as a 2-tuple.
480
481 ```jldoctest
482 julia> extrema(2:10)
483 (2, 10)
484
485 julia> extrema([9,pi,4.5])
486 (3.141592653589793, 9.0)
487 ```
488 """
489 function extrema(itr)
490 s = start(itr)
491 done(itr, s) && throw(ArgumentError("collection must be non-empty"))
492 (v, s) = next(itr, s)
493 vmin = vmax = v
494 while !done(itr, s)
495 (x, s) = next(itr, s)
496 vmax = max(x, vmax)
497 vmin = min(x, vmin)
498 end
499 return (vmin, vmax)
500 end
