256 function copy!(A::SparseMatrixCSC, B::SparseMatrixCSC)
257 # If the two matrices have the same length then all the
258 # elements in A will be overwritten.
259 if length(A) == length(B)
260 resize!(A.nzval, length(B.nzval))
261 resize!(A.rowval, length(B.rowval))
262 if size(A) == size(B)
263 # Simple case: we can simply copy the internal fields of B to A.
264 copy!(A.colptr, B.colptr)
265 copy!(A.rowval, B.rowval)
266 else
267 # This is like a "reshape B into A".
268 sparse_compute_reshaped_colptr_and_rowval(A.colptr, A.rowval, A.m, A.n, B.colptr, B.rowval, B.m, B.n)
269 end
270 else
271 length(A) >= length(B) || throw(BoundsError())
272 lB = length(B)
273 nnzA = nnz(A)
274 nnzB = nnz(B)
275 # Up to which col, row, and ptr in rowval/nzval will A be overwritten?
276 lastmodcolA = div(lB - 1, A.m) + 1
277 lastmodrowA = mod(lB - 1, A.m) + 1
278 lastmodptrA = A.colptr[lastmodcolA]
279 while lastmodptrA < A.colptr[lastmodcolA+1] && A.rowval[lastmodptrA] <= lastmodrowA
280 lastmodptrA += 1
281 end
282 lastmodptrA -= 1
283 if lastmodptrA >= nnzB
284 # A will have fewer non-zero elements; unmodified elements are kept at the end.
285 deleteat!(A.rowval, nnzB+1:lastmodptrA)
286 deleteat!(A.nzval, nnzB+1:lastmodptrA)
287 else
288 # A will have more non-zero elements; unmodified elements are kept at the end.
289 resize!(A.rowval, nnzB + nnzA - lastmodptrA)
290 resize!(A.nzval, nnzB + nnzA - lastmodptrA)
291 copy!(A.rowval, nnzB+1, A.rowval, lastmodptrA+1, nnzA-lastmodptrA)
292 copy!(A.nzval, nnzB+1, A.nzval, lastmodptrA+1, nnzA-lastmodptrA)
293 end
294 # Adjust colptr accordingly.
295 @inbounds for i in 2:length(A.colptr)
296 A.colptr[i] += nnzB - lastmodptrA
297 end
298 sparse_compute_reshaped_colptr_and_rowval(A.colptr, A.rowval, A.m, lastmodcolA-1, B.colptr, B.rowval, B.m, B.n)
299 end
300 copy!(A.nzval, B.nzval)
301 return A
302 end

528 """
529 sparse!{Tv,Ti<:Integer}(
530 I::AbstractVector{Ti}, J::AbstractVector{Ti}, V::AbstractVector{Tv},
531 m::Integer, n::Integer, combine, klasttouch::Vector{Ti},
532 csrrowptr::Vector{Ti}, csrcolval::Vector{Ti}, csrnzval::Vector{Tv},
533 [csccolptr::Vector{Ti}], [cscrowval::Vector{Ti}, cscnzval::Vector{Tv}] )
534
535 Parent of and expert driver for [`sparse`](@ref);
536 see [`sparse`](@ref) for basic usage. This method
537 allows the user to provide preallocated storage for `sparse`'s intermediate objects and
538 result as described below. This capability enables more efficient successive construction
539 of [`SparseMatrixCSC`](@ref)s from coordinate representations, and also enables extraction
540 of an unsorted-column representation of the result's transpose at no additional cost.
541
542 This method consists of three major steps: (1) Counting-sort the provided coordinate
543 representation into an unsorted-row CSR form including repeated entries. (2) Sweep through
544 the CSR form, simultaneously calculating the desired CSC form's column-pointer array,
545 detecting repeated entries, and repacking the CSR form with repeated entries combined;
546 this stage yields an unsorted-row CSR form with no repeated entries. (3) Counting-sort the
547 preceding CSR form into a fully-sorted CSC form with no repeated entries.
548
549 Input arrays `csrrowptr`, `csrcolval`, and `csrnzval` constitute storage for the
550 intermediate CSR forms and require `length(csrrowptr) >= m + 1`,
551 `length(csrcolval) >= length(I)`, and `length(csrnzval >= length(I))`. Input
552 array `klasttouch`, workspace for the second stage, requires `length(klasttouch) >= n`.
553 Optional input arrays `csccolptr`, `cscrowval`, and `cscnzval` constitute storage for the
554 returned CSC form `S`. `csccolptr` requires `length(csccolptr) >= n + 1`. If necessary,
555 `cscrowval` and `cscnzval` are automatically resized to satisfy
556 `length(cscrowval) >= nnz(S)` and `length(cscnzval) >= nnz(S)`; hence, if `nnz(S)` is
557 unknown at the outset, passing in empty vectors of the appropriate type (`Vector{Ti}()`
558 and `Vector{Tv}()` respectively) suffices, or calling the `sparse!` method
559 neglecting `cscrowval` and `cscnzval`.
560
561 On return, `csrrowptr`, `csrcolval`, and `csrnzval` contain an unsorted-column
562 representation of the result's transpose.
563
564 You may reuse the input arrays' storage (`I`, `J`, `V`) for the output arrays
565 (`csccolptr`, `cscrowval`, `cscnzval`). For example, you may call
566 `sparse!(I, J, V, csrrowptr, csrcolval, csrnzval, I, J, V)`.
567
568 For the sake of efficiency, this method performs no argument checking beyond
569 `1 <= I[k] <= m` and `1 <= J[k] <= n`. Use with care. Testing with `--check-bounds=yes`
570 is wise.
571
572 This method runs in `O(m, n, length(I))` time. The HALFPERM algorithm described in
573 F. Gustavson, "Two fast algorithms for sparse matrices: multiplication and permuted
574 transposition," ACM TOMS 4(3), 250-269 (1978) inspired this method's use of a pair of
575 counting sorts.
576 """
577 function sparse!(I::AbstractVector{Ti}, J::AbstractVector{Ti},
578 V::AbstractVector{Tv}, m::Integer, n::Integer, combine, klasttouch::Vector{Ti},
579 csrrowptr::Vector{Ti}, csrcolval::Vector{Ti}, csrnzval::Vector{Tv},
580 csccolptr::Vector{Ti}, cscrowval::Vector{Ti}, cscnzval::Vector{Tv}) where {Tv,Ti<:Integer}
581
582 # Compute the CSR form's row counts and store them shifted forward by one in csrrowptr
583 fill!(csrrowptr, 0)
584 coolen = length(I)
585 @inbounds for k in 1:coolen
586 Ik = I[k]
587 if 1 > Ik || m < Ik
588 throw(ArgumentError("row indices I[k] must satisfy 1 <= I[k] <= m"))
589 end
590 csrrowptr[Ik+1] += 1
591 end
592
593 # Compute the CSR form's rowptrs and store them shifted forward by one in csrrowptr
594 countsum = 1
595 csrrowptr[1] = 1
596 @inbounds for i in 2:(m+1)
597 overwritten = csrrowptr[i]
598 csrrowptr[i] = countsum
599 countsum += overwritten
600 end
601
602 # Counting-sort the column and nonzero values from J and V into csrcolval and csrnzval
603 # Tracking write positions in csrrowptr corrects the row pointers
604 @inbounds for k in 1:coolen
605 Ik, Jk = I[k], J[k]
606 if 1 > Jk || n < Jk
607 throw(ArgumentError("column indices J[k] must satisfy 1 <= J[k] <= n"))
608 end
609 csrk = csrrowptr[Ik+1]
610 csrrowptr[Ik+1] = csrk+1
611 csrcolval[csrk] = Jk
612 csrnzval[csrk] = V[k]
613 end
614 # This completes the unsorted-row, has-repeats CSR form's construction
615
616 # Sweep through the CSR form, simultaneously (1) caculating the CSC form's column
617 # counts and storing them shifted forward by one in csccolptr; (2) detecting repeated
618 # entries; and (3) repacking the CSR form with the repeated entries combined.
619 #
620 # Minimizing extraneous communication and nonlocality of reference, primarily by using
621 # only a single auxiliary array in this step, is the key to this method's performance.
622 fill!(csccolptr, 0)
623 fill!(klasttouch, 0)
624 writek = 1
625 newcsrrowptri = 1
626 origcsrrowptri = 1
627 origcsrrowptrip1 = csrrowptr[2]
628 @inbounds for i in 1:m
629 for readk in origcsrrowptri:(origcsrrowptrip1-1)
630 j = csrcolval[readk]
631 if klasttouch[j] < newcsrrowptri
632 klasttouch[j] = writek
633 if writek != readk
634 csrcolval[writek] = j
635 csrnzval[writek] = csrnzval[readk]
636 end
637 writek += 1
638 csccolptr[j+1] += 1
639 else
640 klt = klasttouch[j]
641 csrnzval[klt] = combine(csrnzval[klt], csrnzval[readk])
642 end
643 end
644 newcsrrowptri = writek
645 origcsrrowptri = origcsrrowptrip1
646 origcsrrowptrip1 != writek && (csrrowptr[i+1] = writek)
647 i < m && (origcsrrowptrip1 = csrrowptr[i+2])
648 end
649
650 # Compute the CSC form's colptrs and store them shifted forward by one in csccolptr
651 countsum = 1
652 csccolptr[1] = 1
653 @inbounds for j in 2:(n+1)
654 overwritten = csccolptr[j]
655 csccolptr[j] = countsum
656 countsum += overwritten
657 end
658
659 # Now knowing the CSC form's entry count, resize cscrowval and cscnzval if necessary
660 cscnnz = countsum - 1
661 length(cscrowval) < cscnnz && resize!(cscrowval, cscnnz)
662 length(cscnzval) < cscnnz && resize!(cscnzval, cscnnz)
663
664 # Finally counting-sort the row and nonzero values from the CSR form into cscrowval and
665 # cscnzval. Tracking write positions in csccolptr corrects the column pointers.
666 @inbounds for i in 1:m
667 for csrk in csrrowptr[i]:(csrrowptr[i+1]-1)
668 j = csrcolval[csrk]
669 x = csrnzval[csrk]
670 csck = csccolptr[j+1]
671 csccolptr[j+1] = csck+1
672 cscrowval[csck] = i
673 cscnzval[csck] = x
674 end
675 end

678 end

727 """
728 halfperm!{Tv,Ti}(X::SparseMatrixCSC{Tv,Ti}, A::SparseMatrixCSC{Tv,Ti},
729 q::AbstractVector{<:Integer}, f::Function = identity)
730
731 Column-permute and transpose `A`, simultaneously applying `f` to each entry of `A`, storing
732 the result `(f(A)Q)^T` (`map(f, transpose(A[:,q]))`) in `X`.
733
734 `X`'s dimensions must match those of `transpose(A)` (`X.m == A.n` and `X.n == A.m`), and `X`
735 must have enough storage to accommodate all allocated entries in `A` (`length(X.rowval) >= nnz(A)`
736 and `length(X.nzval) >= nnz(A)`). Column-permutation `q`'s length must match `A`'s column
737 count (`length(q) == A.n`).
738
739 This method is the parent of several methods performing transposition and permutation
740 operations on [`SparseMatrixCSC`](@ref)s. As this method performs no argument checking,
741 prefer the safer child methods (`[c]transpose[!]`, `permute[!]`) to direct use.
742
743 This method implements the `HALFPERM` algorithm described in F. Gustavson, "Two fast
744 algorithms for sparse matrices: multiplication and permuted transposition," ACM TOMS 4(3),
745 250-269 (1978). The algorithm runs in `O(A.m, A.n, nnz(A))` time and requires no space
746 beyond that passed in.
747 """
748 function halfperm!(X::SparseMatrixCSC{Tv,Ti}, A::SparseMatrixCSC{Tv,Ti},
749 q::AbstractVector{<:Integer}, f::Function = identity) where {Tv,Ti}
750 _computecolptrs_halfperm!(X, A)
751 _distributevals_halfperm!(X, A, q, f)
752 return X
753 end
754 """
755 Helper method for `halfperm!`. Computes `transpose(A[:,q])`'s column pointers, storing them
756 shifted one position forward in `X.colptr`; `_distributevals_halfperm!` fixes this shift.
757 """
758 function _computecolptrs_halfperm!(X::SparseMatrixCSC{Tv,Ti}, A::SparseMatrixCSC{Tv,Ti}) where {Tv,Ti}
759 # Compute `transpose(A[:,q])`'s column counts. Store shifted forward one position in X.colptr.
760 fill!(X.colptr, 0)
761 @inbounds for k in 1:nnz(A)
762 X.colptr[A.rowval[k] + 1] += 1
763 end
764 # Compute `transpose(A[:,q])`'s column pointers. Store shifted forward one position in X.colptr.
765 X.colptr[1] = 1
766 countsum = 1
767 @inbounds for k in 2:(A.m + 1)
768 overwritten = X.colptr[k]
769 X.colptr[k] = countsum
770 countsum += overwritten
771 end
772 end
773 """
774 Helper method for `halfperm!`. With `transpose(A[:,q])`'s column pointers shifted one
775 position forward in `X.colptr`, computes `map(f, transpose(A[:,q]))` by appropriately
776 distributing `A.rowval` and `f`-transformed `A.nzval` into `X.rowval` and `X.nzval`
777 respectively. Simultaneously fixes the one-position-forward shift in `X.colptr`.
778 """
779 function _distributevals_halfperm!(X::SparseMatrixCSC{Tv,Ti},
780 A::SparseMatrixCSC{Tv,Ti}, q::AbstractVector{<:Integer}, f::Function) where {Tv,Ti}
781 @inbounds for Xi in 1:A.n
782 Aj = q[Xi]
783 for Ak in nzrange(A, Aj)
784 Ai = A.rowval[Ak]
785 Xk = X.colptr[Ai + 1]
786 X.rowval[Xk] = Xi
787 X.nzval[Xk] = f(A.nzval[Ak])
788 X.colptr[Ai + 1] += 1
789 end
790 end
791 return # kill potential type instability
792 end

1114 """
1115 fkeep!(A::AbstractSparseArray, f, trim::Bool = true)
1116
1117 Keep elements of `A` for which test `f` returns `true`. `f`'s signature should be
1118
1119 f(i::Integer, [j::Integer,] x) -> Bool
1120
1121 where `i` and `j` are an element's row and column indices and `x` is the element's
1122 value. This method makes a single sweep
1123 through `A`, requiring `O(A.n, nnz(A))`-time for matrices and `O(nnz(A))`-time for vectors
1124 and no space beyond that passed in. If `trim` is `true`, this method trims `A.rowval` or `A.nzind` and
1125 `A.nzval` to length `nnz(A)` after dropping elements.
1126
1127 # Example
1128 ```jldoctest
1129 julia> A = spdiagm([1, 2, 3, 4])
1130 4×4 SparseMatrixCSC{Int64,Int64} with 4 stored entries:
1131 [1, 1] = 1
1132 [2, 2] = 2
1133 [3, 3] = 3
1134 [4, 4] = 4
1135
1136 julia> Base.SparseArrays.fkeep!(A, (i, j, v) -> isodd(v))
1137 4×4 SparseMatrixCSC{Int64,Int64} with 2 stored entries:
1138 [1, 1] = 1
1139 [3, 3] = 3
1140 ```
1141 """
1142 function fkeep!(A::SparseMatrixCSC, f, trim::Bool = true)
1143 An = A.n
1144 Acolptr = A.colptr
1145 Arowval = A.rowval
1146 Anzval = A.nzval
1147
1148 # Sweep through columns, rewriting kept elements in their new positions
1149 # and updating the column pointers accordingly as we go.
1150 Awritepos = 1
1151 oldAcolptrAj = 1
1152 @inbounds for Aj in 1:An
1153 for Ak in oldAcolptrAj:(Acolptr[Aj+1]-1)
1154 Ai = Arowval[Ak]
1155 Ax = Anzval[Ak]
1156 # If this element should be kept, rewrite in new position
1157 if f(Ai, Aj, Ax)
1158 if Awritepos != Ak
1159 Arowval[Awritepos] = Ai
1160 Anzval[Awritepos] = Ax
1161 end
1162 Awritepos += 1
1163 end
1164 end
1165 oldAcolptrAj = Acolptr[Aj+1]
1166 Acolptr[Aj+1] = Awritepos
1167 end
1168
1169 # Trim A's storage if necessary and desired
1170 if trim
1171 Annz = Acolptr[end] - 1
1172 if length(Arowval) != Annz
1173 resize!(Arowval, Annz)
1174 end
1175 if length(Anzval) != Annz
1176 resize!(Anzval, Annz)
1177 end
1178 end
1179
1180 A
1181 end

1293 function sprand_IJ(r::AbstractRNG, m::Integer, n::Integer, density::AbstractFloat)
1294 ((m < 0) || (n < 0)) && throw(ArgumentError("invalid Array dimensions"))
1295 0 <= density <= 1 || throw(ArgumentError("$density not in [0,1]"))
1296 N = n*m
1297
1298 I, J = Vector{Int}(0), Vector{Int}(0) # indices of nonzero elements
1299 sizehint!(I, round(Int,N*density))
1300 sizehint!(J, round(Int,N*density))
1301
1302 # density of nonzero columns:
1303 L = log1p(-density)
1304 coldensity = -expm1(m*L) # = 1 - (1-density)^m
1305 colsparsity = exp(m*L) # = 1 - coldensity
1306 iL = 1/L
1307
1308 rows = Vector{Int}(0)
1309 for j in randsubseq(r, 1:n, coldensity)
1310 # To get the right statistics, we *must* have a nonempty column j
1311 # even if p*m << 1. To do this, we use an approach similar to
1312 # the one in randsubseq to compute the expected first nonzero row k,
1313 # except given that at least one is nonzero (via Bayes' rule);
1314 # carefully rearranged to avoid excessive roundoff errors.
1315 k = ceil(log(colsparsity + rand(r)*coldensity) * iL)
1316 ik = k < 1 ? 1 : k > m ? m : Int(k) # roundoff-error/underflow paranoia
1317 randsubseq!(r, rows, 1:m-ik, density)
1318 push!(rows, m-ik+1)
1319 append!(I, rows)
1320 nrows = length(rows)
1321 Jlen = length(J)
1322 resize!(J, Jlen+nrows)
1323 @inbounds for i = Jlen+1:length(J)
1324 J[i] = j
1325 end
1326 end
1327 I, J
1328 end

1545 ## full equality
1546 function ==(A1::SparseMatrixCSC, A2::SparseMatrixCSC)
1547 size(A1)!=size(A2) && return false
1548 vals1, vals2 = nonzeros(A1), nonzeros(A2)
1549 rows1, rows2 = rowvals(A1), rowvals(A2)
1550 m, n = size(A1)
1551 @inbounds for i = 1:n
1552 nz1,nz2 = nzrange(A1,i), nzrange(A2,i)
1553 j1,j2 = first(nz1), first(nz2)
1554 # step through the rows of both matrices at once:
1555 while j1<=last(nz1) && j2<=last(nz2)
1556 r1,r2 = rows1[j1], rows2[j2]
1557 if r1==r2
1558 vals1[j1]!=vals2[j2] && return false
1559 j1+=1
1560 j2+=1
1561 else
1562 if r1<r2
1563 vals1[j1]!=0 && return false
1564 j1+=1
1565 else
1566 vals2[j2]!=0 && return false
1567 j2+=1
1568 end
1569 end
1570 end
1571 # finish off any left-overs:
1572 for j = j1:last(nz1)
1573 vals1[j]!=0 && return false
1574 end
1575 for j = j2:last(nz2)
1576 vals2[j]!=0 && return false
1577 end
1578 end
1579 return true
1580 end

1592 # General mapreduce
1593 function _mapreducezeros(f, op, ::Type{T}, nzeros::Int, v0) where T
1594 nzeros == 0 && return v0
1595
1596 # Reduce over first zero
1597 zeroval = f(zero(T))
1598 v = op(v0, zeroval)
1599 isequal(v, v0) && return v
1600
1601 # Reduce over remaining zeros
1602 for i = 2:nzeros
1603 lastv = v
1604 v = op(v, zeroval)
1605 # Bail out early if we reach a fixed point
1606 isequal(v, lastv) && break
1607 end
1608
1609 v
1610 end

1679 function Base._mapreducedim!{T}(f, op, R::AbstractArray, A::SparseMatrixCSC{T})
1680 lsiz = Base.check_reducedims(R,A)
1681 isempty(A) && return R
1682
1683 if size(R, 1) == size(R, 2) == 1
1684 # Reduction along both columns and rows
1685 R[1, 1] = mapreduce(f, op, A)
1686 elseif size(R, 1) == 1
1687 # Reduction along rows
1688 _mapreducerows!(f, op, R, A)
1689 elseif size(R, 2) == 1
1690 # Reduction along columns
1691 _mapreducecols!(f, op, R, A)
1692 else
1693 # Reduction along a dimension > 2
1694 # Compute op(R, f(A))
1695 m, n = size(A)
1696 nzval = A.nzval
1697 if length(nzval) == m*n
1698 # No zeros, so don't compute f(0) since it might throw
1699 for col = 1:n
1700 @simd for row = 1:size(A, 1)
1701 @inbounds R[row, col] = op(R[row, col], f(nzval[(col-1)*m+row]))
1702 end
1703 end
1704 else
1705 colptr = A.colptr
1706 rowval = A.rowval
1707 zeroval = f(zero(T))
1708 @inbounds for col = 1:n
1709 lastrow = 0
1710 for j = colptr[col]:colptr[col+1]-1
1711 row = rowval[j]
1712 @simd for i = lastrow+1:row-1 # Zeros before this nonzero
1713 R[i, col] = op(R[i, col], zeroval)
1714 end
1715 R[row, col] = op(R[row, col], f(nzval[j]))
1716 lastrow = row
1717 end
1718 @simd for i = lastrow+1:m # Zeros at end
1719 R[i, col] = op(R[i, col], zeroval)
1720 end
1721 end
1722 end
1723 end
1724 R
1725 end

1768 # findmax/min and indmax/min methods
1769 function _findz{Tv,Ti}(A::SparseMatrixCSC{Tv,Ti}, rows=1:A.m, cols=1:A.n)
1770 colptr = A.colptr; rowval = A.rowval; nzval = A.nzval
1771 zval = zero(Tv)
1772 col = cols[1]; row = 0
1773 rowmin = rows[1]; rowmax = rows[end]
1774 allrows = (rows == 1:A.m)
1775 @inbounds while col <= cols[end]
1776 r1::Int = colptr[col]
1777 r2::Int = colptr[col+1] - 1
1778 if !allrows && (r1 <= r2)
1779 r1 = searchsortedfirst(rowval, rowmin, r1, r2, Forward)
1780 (r1 <= r2 ) && (r2 = searchsortedlast(rowval, rowmax, r1, r2, Forward) + 1)
1781 end
1782 row = rowmin
1783 (r1 > r2) && (return sub2ind(size(A),row,col))
1784 while (r1 <= r2) && (row == rowval[r1]) && (nzval[r1] != zval)
1785 r1 += 1
1786 row += 1
1787 end
1788 (row <= rowmax) && (return sub2ind(size(A),row,col))
1789 col += 1
1790 end
1791 return 0
1792 end
