35 # LLVM has several code generation bugs for checked integer arithmetic (see e.g.
36 # #4905). We thus distinguish between operations that can be implemented via
37 # intrinsics, and operations for which we have to provide work-arounds.
38
39 # Note: As far as this code has been tested, most checked_* functions are
40 # working fine in LLVM. (Note that division is still handled via `base/int.jl`,
41 # which always checks for overflow, and which provides its own sets of
42 # work-arounds for LLVM codegen bugs.) However, the comments in `base/int.jl`
43 # and in issue #4905 are more pessimistic. For the time being, we thus retain
44 # the ability to handle codegen bugs in LLVM, until the code here has been
45 # tested on more systems and architectures. It also seems that things depend on
46 # which compiler that was used to build LLVM (i.e. either gcc or clang).
47

49 # BrokenSignedInt
50 # BrokenUnsignedInt

52 # BrokenSignedIntMul
53 # BrokenUnsignedIntMul

81 """
82 Base.checked_neg(x)
83
84 Calculates `-x`, checking for overflow errors where applicable. For
85 example, standard two's complement signed integers (e.g. `Int`) cannot
86 represent `-typemin(Int)`, thus leading to an overflow.
87
88 The overflow protection may impose a perceptible performance penalty.
89 """
90 function checked_neg(x::T) where T<:Integer
91 checked_sub(T(0), x)
92 end

95 r = -x

97 r
98 end
99 end

101 function checked_neg(x::T) where T<:BrokenUnsignedInt
102 x != 0 && throw(OverflowError())
103 T(0)
104 end

251 if Int128 <: BrokenSignedIntMul
252 # Avoid BigInt
253 function mul_with_overflow(x::T, y::T) where T<:Int128
254 f = if y > 0
255 # x * y > typemax(T)
256 # x * y < typemin(T)
257 x > fld(typemax(T), y) || x < cld(typemin(T), y)
258 elseif y < 0
259 # x * y > typemax(T)
260 # x * y < typemin(T)
261 # y == -1 can overflow fld
262 x < cld(typemax(T), y) || y != -1 && x > fld(typemin(T), y)
263 else
264 false
265 end
266 x*y, f
267 end
268 end

270 # Avoid BigInt
271 function mul_with_overflow(x::T, y::T) where T<:UInt128
272 # x * y > typemax(T)
273 x * y, y > 0 && x > fld(typemax(T), y)
274 end
275 end
