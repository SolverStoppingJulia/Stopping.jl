190 function transcode(::Type{UInt16}, src::Vector{UInt8})
191 dst = UInt16[]
192 i, n = 1, length(src)
193 n > 0 || return dst
194 sizehint!(dst, 2n)
195 a = src[1]
196 while true
197 if i < n && -64 <= a % Int8 <= -12 # multi-byte character
198 b = src[i += 1]
199 if -64 <= (b % Int8) || a == 0xf4 && 0x8f < b
200 # invalid UTF-8 (non-continuation or too-high code point)
201 push!(dst, a)
202 a = b; continue
203 elseif a < 0xe0 # 2-byte UTF-8
204 push!(dst, xor(0x3080, UInt16(a) << 6, b))
205 elseif i < n # 3/4-byte character
206 c = src[i += 1]
207 if -64 <= (c % Int8) # invalid UTF-8 (non-continuation)
208 push!(dst, a, b)
209 a = c; continue
210 elseif a < 0xf0 # 3-byte UTF-8
211 push!(dst, xor(0x2080, UInt16(a) << 12, UInt16(b) << 6, c))
212 elseif i < n
213 d = src[i += 1]
214 if -64 <= (d % Int8) # invalid UTF-8 (non-continuation)
215 push!(dst, a, b, c)
216 a = d; continue
217 elseif a == 0xf0 && b < 0x90 # overlong encoding
218 push!(dst, xor(0x2080, UInt16(b) << 12, UInt16(c) << 6, d))
219 else # 4-byte UTF-8
220 push!(dst, 0xe5b8 + (UInt16(a) << 8) + (UInt16(b) << 2) + (c >> 4),
221 xor(0xdc80, UInt16(c & 0xf) << 6, d))
222 end
223 else # too short
224 push!(dst, a, b, c)
225 break
226 end
227 else # too short
228 push!(dst, a, b)
229 break
230 end
231 else # ASCII or invalid UTF-8 (continuation byte or too-high code point)
232 push!(dst, a)
233 end
234 i < n || break
235 a = src[i += 1]
236 end
237 return dst
238 end

240 function transcode(::Type{UInt8}, src::Vector{UInt16})
241 n = length(src)
242 n == 0 && return UInt8[]
243
244 # Precompute m = sizeof(dst). This involves annoying duplication
245 # of the loop over the src array. However, this is not just an
246 # optimization: it is problematic for security reasons to grow
247 # dst dynamically, because Base.winprompt uses this function to
248 # convert passwords to UTF-8 and we don't want to make unintentional
249 # copies of the password data.
250 a = src[1]
251 i, m = 1, 0
252 while true
253 if a < 0x80
254 m += 1
255 elseif a < 0x800 # 2-byte UTF-8
256 m += 2
257 elseif a & 0xfc00 == 0xd800 && i < length(src)
258 b = src[i += 1]
259 if (b & 0xfc00) == 0xdc00 # 2-unit UTF-16 sequence => 4-byte UTF-8
260 m += 4
261 else
262 m += 3
263 a = b; continue
264 end
265 else
266 # 1-unit high UTF-16 or unpaired high surrogate
267 # either way, encode as 3-byte UTF-8 code point
268 m += 3
269 end
270 i < n || break
271 a = src[i += 1]
272 end
273
274 dst = StringVector(m)
275 a = src[1]
276 i, j = 1, 0
277 while true
278 if a < 0x80 # ASCII
279 dst[j += 1] = a % UInt8
280 elseif a < 0x800 # 2-byte UTF-8
281 dst[j += 1] = 0xc0 | ((a >> 6) % UInt8)
282 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
283 elseif a & 0xfc00 == 0xd800 && i < n
284 b = src[i += 1]
285 if (b & 0xfc00) == 0xdc00
286 # 2-unit UTF-16 sequence => 4-byte UTF-8
287 a += 0x2840
288 dst[j += 1] = 0xf0 | ((a >> 8) % UInt8)
289 dst[j += 1] = 0x80 | ((a % UInt8) >> 2)
290 dst[j += 1] = xor(0xf0, ((a % UInt8) << 4) & 0x3f, (b >> 6) % UInt8)
291 dst[j += 1] = 0x80 | ((b % UInt8) & 0x3f)
292 else
293 dst[j += 1] = 0xe0 | ((a >> 12) % UInt8)
294 dst[j += 1] = 0x80 | (((a >> 6) % UInt8) & 0x3f)
295 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
296 a = b; continue
297 end
298 else
299 # 1-unit high UTF-16 or unpaired high surrogate
300 # either way, encode as 3-byte UTF-8 code point
301 dst[j += 1] = 0xe0 | ((a >> 12) % UInt8)
302 dst[j += 1] = 0x80 | (((a >> 6) % UInt8) & 0x3f)
303 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
304 end
305 i < n || break
306 a = src[i += 1]
307 end
308 return dst
309 end

321 """
322 disable_sigint(f::Function)
323
324 Disable Ctrl-C handler during execution of a function on the current task,
325 for calling external code that may call julia code that is not interrupt safe.
326 Intended to be called using `do` block syntax as follows:
327
328 disable_sigint() do
329 # interrupt-unsafe code
330 ...
331 end
332
333 This is not needed on worker threads (`Threads.threadid() != 1`) since the
334 `InterruptException` will only be delivered to the master thread.
335 External functions that do not call julia code or julia runtime
336 automatically disable sigint during their execution.
337 """
338 function disable_sigint(f::Function)
339 sigatomic_begin()
340 res = f()
341 # Exception unwind sigatomic automatically
342 sigatomic_end()
343 res
344 end
