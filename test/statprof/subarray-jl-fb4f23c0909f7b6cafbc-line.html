350 Recursively replace occurrences of the symbol :end in a "ref" expression (i.e. A[...]) `ex`

361 replace_ref_end!(ex) = replace_ref_end_!(ex, nothing)[1]
362 # replace_ref_end_!(ex,withex) returns (new ex, whether withex was used)
363 function replace_ref_end_!(ex, withex)
364 used_withex = false
365 if isa(ex,Symbol) && ex == :end
366 withex === nothing && error("Invalid use of end")
367 return withex, true
368 elseif isa(ex,Expr)
369 if ex.head == :ref
370 ex.args[1], used_withex = replace_ref_end_!(ex.args[1],withex)
371 S = isa(ex.args[1],Symbol) ? ex.args[1]::Symbol : gensym(:S) # temp var to cache ex.args[1] if needed
372 used_S = false # whether we actually need S
373 # new :ref, so redefine withex
374 nargs = length(ex.args)-1
375 if nargs == 0
376 return ex, used_withex
377 elseif nargs == 1
378 # replace with endof(S)
379 ex.args[2], used_S = replace_ref_end_!(ex.args[2],:($endof($S)))
380 else
381 n = 1
382 J = endof(ex.args)
383 for j = 2:J-1
384 exj, used = replace_ref_end_!(ex.args[j],:($size($S,$n)))
385 used_S |= used
386 ex.args[j] = exj
387 if isa(exj,Expr) && exj.head == :...
388 # splatted object
389 exjs = exj.args[1]
390 n = :($n + length($exjs))
391 elseif isa(n, Expr)
392 # previous expression splatted
393 n = :($n + 1)
394 else
395 # an integer
396 n += 1
397 end
398 end
399 ex.args[J], used = replace_ref_end_!(ex.args[J],:($trailingsize($S,$n)))
400 used_S |= used
401 end
402 if used_S && S !== ex.args[1]
403 S0 = ex.args[1]
404 ex.args[1] = S
405 ex = Expr(:let, ex, :($S = $S0))
406 end
407 else
408 # recursive search
409 for i = eachindex(ex.args)
410 ex.args[i], used = replace_ref_end_!(ex.args[i],withex)
411 used_withex |= used
412 end
413 end
414 end
415 ex, used_withex
416 end

419 @view A[inds...]
420
421 Creates a `SubArray` from an indexing expression. This can only be applied directly to a
422 reference expression (e.g. `@view A[1,2:end]`), and should *not* be used as the target of
423 an assignment (e.g. `@view(A[1,2:end]) = ...`). See also [`@views`](@ref)
424 to switch an entire block of code to use views for slicing.
425
426 ```jldoctest
427 julia> A = [1 2; 3 4]
428 2√É¬ó2 Array{Int64,2}:
429 1 2
430 3 4
431
432 julia> b = @view A[:, 1]

434 1
435 3
436
437 julia> fill!(b, 0)
438 2-element SubArray{Int64,1,Array{Int64,2},Tuple{Base.Slice{Base.OneTo{Int64}},Int64},true}:
439 0
440 0
441
442 julia> A
443 2√É¬ó2 Array{Int64,2}:
444 0 2
445 0 4

448 macro view(ex)
449 if Meta.isexpr(ex, :ref)
450 ex = replace_ref_end!(ex)
451 if Meta.isexpr(ex, :ref)
452 ex = Expr(:call, view, ex.args...)
453 else # ex replaced by let ...; foo[...]; end
454 assert(Meta.isexpr(ex, :let) && Meta.isexpr(ex.args[1], :ref))
455 ex.args[1] = Expr(:call, view, ex.args[1].args...)
456 end
457 Expr(:&&, true, esc(ex))

474 # _views implements the transformation for the @views macro.
475 # @views calls esc(_views(...)) to work around #20241,
476 # so any function calls we insert (to maybeview, or to
477 # size and endof in replace_ref_end!) must be interpolated
478 # as values rather than as symbols to ensure that they are called
479 # from Base rather than from the caller's scope.
480 _views(x) = x
481 function _views(ex::Expr)
482 if ex.head in (:(=), :(.=))
483 # don't use view for ref on the lhs of an assignment,
484 # but still use views for the args of the ref:
485 lhs = ex.args[1]
486 Expr(ex.head, Meta.isexpr(lhs, :ref) ?
487 Expr(:ref, _views.(lhs.args)...) : _views(lhs),
488 _views(ex.args[2]))
489 elseif ex.head == :ref
490 Expr(:call, maybeview, _views.(ex.args)...)
491 else
492 h = string(ex.head)
493 # don't use view on the lhs of an op-assignment a[i...] += ...
494 if last(h) == '=' && Meta.isexpr(ex.args[1], :ref)
495 lhs = ex.args[1]
496
497 # temp vars to avoid recomputing a and i,
498 # which will be assigned in a let block:
499 a = gensym(:a)
500 i = [gensym(:i) for k = 1:length(lhs.args)-1]
501
502 # for splatted indices like a[i, j...], we need to
503 # splat the corresponding temp var.
504 I = similar(i, Any)
505 for k = 1:length(i)
506 if Meta.isexpr(lhs.args[k+1], :...)
507 I[k] = Expr(:..., i[k])
508 lhs.args[k+1] = lhs.args[k+1].args[1] # unsplat
509 else
510 I[k] = i[k]
511 end
512 end
513
514 Expr(:let,
515 Expr(first(h) == '.' ? :(.=) : :(=), :($a[$(I...)]),
516 Expr(:call, Symbol(h[1:end-1]),
517 :($maybeview($a, $(I...))),
518 _views.(ex.args[2:end])...)),
519 :($a = $(_views(lhs.args[1]))),
520 [:($(i[k]) = $(_views(lhs.args[k+1]))) for k=1:length(i)]...)
521 else
522 Expr(ex.head, _views.(ex.args)...)
523 end
524 end
525 end

527 """
528 @views expression
529
530 Convert every array-slicing operation in the given expression
531 (which may be a `begin`/`end` block, loop, function, etc.)
532 to return a view. Scalar indices, non-array types, and
533 explicit `getindex` calls (as opposed to `array[...]`) are
534 unaffected.
535
536 Note that the `@views` macro only affects `array[...]` expressions
537 that appear explicitly in the given `expression`, not array slicing that
538 occurs in functions called by that code.
539 """
540 macro views(x)
541 esc(_views(replace_ref_end!(x)))
