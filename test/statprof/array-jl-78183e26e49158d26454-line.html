503 function collect_to!(dest::AbstractArray{T}, itr, offs, st) where T
504 # collect to dest array, checking the type of each result. if a result does not
505 # match, widen the result type and re-dispatch.
506 i = offs
507 while !done(itr, st)
508 el, st = next(itr, st)
509 S = typeof(el)
510 if S === T || S <: T
511 @inbounds dest[i] = el::T
512 i += 1
513 else
514 R = typejoin(T, S)
515 new = similar(dest, R)
516 copy!(new,1, dest,1, i-1)
517 @inbounds new[i] = el
518 return collect_to!(new, itr, i+1, st)
519 end
520 end
521 return dest
522 end

916 function _deleteat!(a::Vector, inds)
917 n = length(a)
918 s = start(inds)
919 done(inds, s) && return a
920 (p, s) = next(inds, s)
921 q = p+1
922 while !done(inds, s)
923 (i,s) = next(inds, s)
924 if !(q <= i <= n)
925 if i < q
926 throw(ArgumentError("indices must be unique and sorted"))
927 else
928 throw(BoundsError())
929 end
930 end
931 while q < i
932 @inbounds a[p] = a[q]
933 p += 1; q += 1
934 end
935 q = i+1
936 end
937 while q <= n
938 @inbounds a[p] = a[q]
939 p += 1; q += 1
940 end
941 ccall(:jl_array_del_end, Void, (Any, UInt), a, n-p+1)
942 return a
943 end

1079 k += 1
