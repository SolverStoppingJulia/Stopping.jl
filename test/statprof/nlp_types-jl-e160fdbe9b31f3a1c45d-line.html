219 """
220 has_bounds(nlp)
221 has_bounds(meta)
222
223 Returns whether the problem has bounds on the variables.
224 """
225 has_bounds(meta::NLPModelMeta) = length(meta.ifree) < meta.nvar
226
227 """
228 bound_constrained(nlp)
229 bound_constrained(meta)
230
231 Returns whether the problem has bounds on the variables and no other constraints.
232 """
233 bound_constrained(meta::NLPModelMeta) = meta.ncon == 0 && has_bounds(meta)
234
235 """
236 unconstrained(nlp)
237 unconstrained(meta)
238
239 Returns whether the problem in unconstrained.
240 """
241 unconstrained(meta::NLPModelMeta) = meta.ncon == 0 && !has_bounds(meta)
242
243 """
244 linearly_constrained(nlp)
245 linearly_constrained(meta)
246
247 Returns whether the problem's constraints are known to be all linear.
248 """
249 linearly_constrained(meta::NLPModelMeta) = meta.nlin == meta.ncon > 0
250
251 """
252 equality_constrained(nlp)
253 equality_constrained(meta)
254
255 Returns whether the problem's constraints are all equalities.
256 Unconstrained problems return false.
257 """
258 equality_constrained(meta::NLPModelMeta) = length(meta.jfix) == meta.ncon > 0
259
260 """
261 inequality_constrained(nlp)
262 inequality_constrained(meta)
263
264 Returns whether the problem's constraints are all inequalities.
265 Unconstrained problems return true.
266 """
267 inequality_constrained(meta::NLPModelMeta) = meta.ncon > 0 && length(meta.jfix) == 0
