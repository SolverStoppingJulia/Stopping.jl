499 if Core.sizeof(Int) == 4
500 function widemul(u::Int64, v::Int64)
501 local u0::UInt64, v0::UInt64, w0::UInt64
502 local u1::Int64, v1::Int64, w1::UInt64, w2::Int64, t::UInt64
503
504 u0 = u & 0xffffffff; u1 = u >> 32
505 v0 = v & 0xffffffff; v1 = v >> 32
506 w0 = u0 * v0
507 t = reinterpret(UInt64, u1) * v0 + (w0 >>> 32)
508 w2 = reinterpret(Int64, t) >> 32
509 w1 = u0 * reinterpret(UInt64, v1) + (t & 0xffffffff)
510 hi = u1 * v1 + w2 + (reinterpret(Int64, w1) >> 32)
511 lo = w0 & 0xffffffff + (w1 << 32)
512 return Int128(hi) << 64 + Int128(lo)
513 end

515 function widemul(u::UInt64, v::UInt64)
516 local u0::UInt64, v0::UInt64, w0::UInt64
517 local u1::UInt64, v1::UInt64, w1::UInt64, w2::UInt64, t::UInt64
518
519 u0 = u & 0xffffffff; u1 = u >>> 32
520 v0 = v & 0xffffffff; v1 = v >>> 32
521 w0 = u0 * v0
522 t = u1 * v0 + (w0 >>> 32)
523 w2 = t >>> 32
524 w1 = u0 * v1 + (t & 0xffffffff)
525 hi = u1 * v1 + w2 + (w1 >>> 32)
526 lo = w0 & 0xffffffff + (w1 << 32)
527 return UInt128(hi) << 64 + UInt128(lo)
528 end

530 function *(u::Int128, v::Int128)
531 u0 = u % UInt64; u1 = Int64(u >> 64)
532 v0 = v % UInt64; v1 = Int64(v >> 64)
533 lolo = widemul(u0, v0)
534 lohi = widemul(reinterpret(Int64, u0), v1)
535 hilo = widemul(u1, reinterpret(Int64, v0))
536 t = reinterpret(UInt128, hilo) + (lolo >>> 64)
537 w1 = reinterpret(UInt128, lohi) + (t & 0xffffffffffffffff)
538 return Int128(lolo & 0xffffffffffffffff) + reinterpret(Int128, w1) << 64
539 end
