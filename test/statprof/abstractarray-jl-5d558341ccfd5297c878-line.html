1786 function mapslices(f, A::AbstractArray, dims::AbstractVector)
1787 if isempty(dims)
1788 return map(f,A)
1789 end
1790
1791 dimsA = [indices(A)...]
1792 ndimsA = ndims(A)
1793 alldims = [1:ndimsA;]
1794
1795 otherdims = setdiff(alldims, dims)
1796
1797 idx = Any[first(ind) for ind in indices(A)]
1798 itershape = tuple(dimsA[otherdims]...)
1799 for d in dims
1800 idx[d] = Slice(indices(A, d))
1801 end
1802
1803 # Apply the function to the first slice in order to determine the next steps
1804 Aslice = A[idx...]
1805 r1 = f(Aslice)
1806 # In some cases, we can re-use the first slice for a dramatic performance
1807 # increase. The slice itself must be mutable and the result cannot contain
1808 # any mutable containers. The following errs on the side of being overly
1809 # strict (#18570 & #21123).
1810 safe_for_reuse = isa(Aslice, StridedArray) &&
1811 (isa(r1, Number) || (isa(r1, AbstractArray) && eltype(r1) <: Number))
1812
1813 # determine result size and allocate
1814 Rsize = copy(dimsA)
1815 # TODO: maybe support removing dimensions
1816 if !isa(r1, AbstractArray) || ndims(r1) == 0
1817 r1 = [r1]
1818 end
1819 nextra = max(0, length(dims)-ndims(r1))
1820 if eltype(Rsize) == Int
1821 Rsize[dims] = [size(r1)..., ntuple(d->1, nextra)...]
1822 else
1823 Rsize[dims] = [indices(r1)..., ntuple(d->OneTo(1), nextra)...]
1824 end
1825 R = similar(r1, tuple(Rsize...,))
1826
1827 ridx = Any[map(first, indices(R))...]
1828 for d in dims
1829 ridx[d] = indices(R,d)
1830 end
1831
1832 R[ridx...] = r1
1833
1834 nidx = length(otherdims)
1835 if safe_for_reuse
1836 # when f returns an array, R[ridx...] = f(Aslice) line copies elements,
1837 # so we can reuse Aslice
1838 for I in Iterators.drop(CartesianRange(itershape), 1) # skip the first element, we already handled it
1839 for i in 1:nidx
1840 idx[otherdims[i]] = ridx[otherdims[i]] = I.I[i]
1841 end
1842 _unsafe_getindex!(Aslice, A, idx...)
1843 R[ridx...] = f(Aslice)
1844 end
1845 else
1846 # we can't guarantee safety (#18524), so allocate new storage for each slice
1847 for I in Iterators.drop(CartesianRange(itershape), 1)
1848 for i in 1:nidx
1849 idx[otherdims[i]] = ridx[otherdims[i]] = I.I[i]
1850 end
1851 R[ridx...] = f(A[idx...])
1852 end
1853 end
1854
1855 return R
1856 end

1947 const hashaa_seed = UInt === UInt64 ? 0x7f53e68ceb575e76 : 0xeb575e76
1948 const hashrle_seed = UInt === UInt64 ? 0x2aab8909bfea414c : 0xbfea414c
1949 function hash(a::AbstractArray, h::UInt)
1950 h += hashaa_seed
1951 h += hash(size(a))
1952
1953 state = start(a)
1954 done(a, state) && return h
1955 x2, state = next(a, state)
1956 done(a, state) && return hash(x2, h)
1957
1958 x1 = x2
1959 while !done(a, state)
1960 x1 = x2
1961 x2, state = next(a, state)
1962 if isequal(x2, x1)
1963 # For repeated elements, use run length encoding
1964 # This allows efficient hashing of sparse arrays
1965 runlength = 2
1966 while !done(a, state)
1967 x2, state = next(a, state)
1968 isequal(x1, x2) || break
1969 runlength += 1
1970 end
1971 h += hashrle_seed
1972 h = hash(runlength, h)
1973 end
1974 h = hash(x1, h)
1975 end
1976 !isequal(x2, x1) && (h = hash(x2, h))
1977 return h
