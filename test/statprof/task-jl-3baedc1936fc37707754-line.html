155 """
156 task_local_storage(body, key, value)
157
158 Call the function `body` with a modified task-local storage, in which `value` is assigned to
159 `key`; the previous value of `key`, or lack thereof, is restored afterwards. Useful
160 for emulating dynamic scoping.
161 """
162 function task_local_storage(body::Function, key, val)
163 tls = task_local_storage()
164 hadkey = haskey(tls,key)
165 old = get(tls,key,nothing)
166 tls[key] = val
167 try body()
168 finally
169 hadkey ? (tls[key] = old) : delete!(tls,key)
170 end
171 end

174 function wait(t::Task)
175 if !istaskdone(t)
176 if t.donenotify === nothing
177 t.donenotify = Condition()
178 end
179 end
180 while !istaskdone(t)
181 wait(t.donenotify)
182 end
183 if istaskfailed(t)
184 throw(t.exception)
185 end
186 return task_result(t)
187 end

341 timedwait(testcb::Function, secs::Float64; pollint::Float64=0.1)
342
343 Waits until `testcb` returns `true` or for `secs` seconds, whichever is earlier.
344 `testcb` is polled every `pollint` seconds.
345 """
346 function timedwait(testcb::Function, secs::Float64; pollint::Float64=0.1)
347 pollint > 0 || throw(ArgumentError("cannot set pollint to $pollint seconds"))
348 start = time()
349 done = Channel(1)
350 timercb(aw) = begin
351 try
352 if testcb()
353 put!(done, :ok)
354 elseif (time() - start) > secs
355 put!(done, :timed_out)
356 end
357 catch e
358 put!(done, :error)
359 finally
360 isready(done) && close(aw)
361 end
362 end
363
364 if !testcb()
365 t = Timer(timercb, pollint, pollint)
366 ret = fetch(done)
367 close(t)
368 else
369 ret = :ok
370 end
371 ret
372 end
