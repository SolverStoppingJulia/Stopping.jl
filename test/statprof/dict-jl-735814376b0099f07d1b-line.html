177 function grow_to!(dest::Associative{K,V}, itr, st) where V where K
178 while !done(itr, st)
179 (k,v), st = next(itr, st)
180 if isa(k,K) && isa(v,V)
181 dest[k] = v
182 else
183 new = similar(dest, Pair{typejoin(K,typeof(k)), typejoin(V,typeof(v))})
184 copy!(new, dest)
185 new[k] = v
186 return grow_to!(new, itr, st)
187 end
188 end
189 return dest

216 function rehash!(h::Dict{K,V}, newsz = length(h.keys)) where V where K
217 olds = h.slots
218 oldk = h.keys
219 oldv = h.vals
220 sz = length(olds)
221 newsz = _tablesz(newsz)
222 h.age += 1
223 h.idxfloor = 1
224 if h.count == 0
225 resize!(h.slots, newsz)
226 fill!(h.slots, 0)
227 resize!(h.keys, newsz)
228 resize!(h.vals, newsz)
229 h.ndel = 0
230 return h
231 end
232
233 slots = zeros(UInt8,newsz)
234 keys = Array{K,1}(newsz)
235 vals = Array{V,1}(newsz)
236 age0 = h.age
237 count = 0
238 maxprobe = h.maxprobe
239
240 for i = 1:sz
241 if olds[i] == 0x1
242 k = oldk[i]
243 v = oldv[i]
244 index0 = index = hashindex(k, newsz)
245 while slots[index] != 0
246 index = (index & (newsz-1)) + 1
247 end
248 probe = (index - index0) & (newsz-1)
249 probe > maxprobe && (maxprobe = probe)
250 slots[index] = 0x1
251 keys[index] = k
252 vals[index] = v
253 count += 1
254
255 if h.age != age0
256 # if `h` is changed by a finalizer, retry
257 return rehash!(h, newsz)
258 end
259 end
260 end
261
262 h.slots = slots
263 h.keys = keys
264 h.vals = vals
265 h.count = count
266 h.ndel = 0
267 h.maxprobe = maxprobe
268 @assert h.age == age0
269
270 return h
271 end

318 function ht_keyindex(h::Dict{K,V}, key) where V where K
319 sz = length(h.keys)
320 iter = 0
321 maxprobe = h.maxprobe
322 index = hashindex(key, sz)
323 keys = h.keys
324
325 while true
326 if isslotempty(h,index)
327 break
328 end
329 if !isslotmissing(h,index) && (key === keys[index] || isequal(key,keys[index]))
330 return index
331 end
332
333 index = (index & (sz-1)) + 1
334 iter += 1
335 iter > maxprobe && break
336 end
337 return -1
338 end

340 # get the index where a key is stored, or -pos if not present
341 # and the key would be inserted at pos
342 # This version is for use by setindex! and get!
343 function ht_keyindex2(h::Dict{K,V}, key) where V where K
344 age0 = h.age
345 sz = length(h.keys)
346 iter = 0
347 maxprobe = h.maxprobe
348 index = hashindex(key, sz)
349 avail = 0
350 keys = h.keys
351
352 while true
353 if isslotempty(h,index)
354 if avail < 0
355 return avail
356 end
357 return -index
358 end
359
360 if isslotmissing(h,index)
361 if avail == 0
362 # found an available slot, but need to keep scanning
363 # in case "key" already exists in a later collided slot.
364 avail = -index
365 end
366 elseif key === keys[index] || isequal(key, keys[index])
367 return index
368 end
369
370 index = (index & (sz-1)) + 1
371 iter += 1
372 iter > maxprobe && break
373 end
374
375 avail < 0 && return avail
376
377 maxallowed = max(maxallowedprobe, sz>>maxprobeshift)
378 # Check if key is not present, may need to keep searching to find slot
379 while iter < maxallowed
380 if !isslotfilled(h,index)
381 h.maxprobe = iter
382 return -index
383 end
384 index = (index & (sz-1)) + 1
385 iter += 1
386 end
387
388 rehash!(h, h.count > 64000 ? sz*2 : sz*4)
389
390 return ht_keyindex2(h, key)
391 end
