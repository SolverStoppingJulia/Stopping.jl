404 # Summary:
405 # print(io, ex) defers to show_unquoted(io, ex)
406 # show(io, ex) defers to show_unquoted(io, QuoteNode(ex))
407 # show_unquoted(io, ex) does the heavy lifting
408 #
409 # AST printing should follow two rules:
410 # 1. parse(string(ex)) == ex
411 # 2. eval(parse(repr(ex))) == ex
412 #
413 # Rule 1 means that printing an expression should generate Julia code which
414 # could be reparsed to obtain the original expression. This code should be
415 # unambiguous and as readable as possible.
416 #
417 # Rule 2 means that showing an expression should generate a quoted version of
418 # print√¢¬Ä¬ôs output. Parsing and then evaling this output should return the
419 # original expression.
420 #
421 # This is consistent with many other show methods, i.e.:
422 # show(Set([1,2,3])) # ==> "Set{Int64}([2,3,1])"
423 # eval(parse("Set{Int64}([2,3,1])√¢¬Ä¬ù) # ==> An actual set
424 # While this isn√¢¬Ä¬ôt true of ALL show methods, it is of all ASTs.

690 # TODO: implement interpolated strings
691 function show_unquoted(io::IO, ex::Expr, indent::Int, prec::Int)
692 head, args, nargs = ex.head, ex.args, length(ex.args)
693 emphstate = typeemphasize(io)
694 show_type = true
695 if (ex.head == :(=) || ex.head == :line ||
696 ex.head == :boundscheck ||
697 ex.head == :gotoifnot ||
698 ex.head == :return)
699 show_type = false
700 end
701 if !emphstate && ex.typ === Any
702 show_type = false
703 end
704 # dot (i.e. "x.y"), but not compact broadcast exps
705 if head === :(.) && !is_expr(args[2], :tuple)
706 show_unquoted(io, args[1], indent + indent_width)
707 print(io, '.')
708 if is_quoted(args[2])
709 show_unquoted(io, unquoted(args[2]), indent + indent_width)
710 else
711 print(io, '(')
712 show_unquoted(io, args[2], indent + indent_width)
713 print(io, ')')
714 end
715
716 # infix (i.e. "x <: y" or "x = y")
717 elseif (head in expr_infix_any && nargs==2) || (head === :(:) && nargs==3)
718 func_prec = operator_precedence(head)
719 head_ = head in expr_infix_wide ? " $head " : head
720 if func_prec <= prec
721 show_enclosed_list(io, '(', args, head_, ')', indent, func_prec, true)
722 else
723 show_list(io, args, head_, indent, func_prec, true)
724 end
725
726 # list (i.e. "(1, 2, 3)" or "[1, 2, 3]")
727 elseif haskey(expr_parens, head) # :tuple/:vcat
728 op, cl = expr_parens[head]
729 if head === :vcat
730 sep = "; "
731 elseif head === :hcat || head === :row
732 sep = " "
733 else
734 sep = ", "
735 end
736 head !== :row && print(io, op)
737 show_list(io, args, sep, indent)
738 if nargs == 1
739 if head === :tuple
740 print(io, ',')
741 elseif head === :vcat
742 print(io, ';')
743 end
744 end
745 head !== :row && print(io, cl)
746
747 # function call
748 elseif head === :call && nargs >= 1
749 func = args[1]
750 fname = isa(func,GlobalRef) ? func.name : func
751 func_prec = operator_precedence(fname)
752 if func_prec > 0 || fname in uni_ops
753 func = fname
754 end
755 func_args = args[2:end]
756
757 if (in(ex.args[1], (GlobalRef(Base, :bitcast), :throw)) ||
758 ismodulecall(ex))
759 show_type = false
760 end
761 if show_type
762 prec = prec_decl
763 end
764
765 # scalar multiplication (i.e. "100x")
766 if (func === :* &&
767 length(func_args)==2 && isa(func_args[1], Real) && isa(func_args[2], Symbol))
768 if func_prec <= prec
769 show_enclosed_list(io, '(', func_args, "", ')', indent, func_prec)
770 else
771 show_list(io, func_args, "", indent, func_prec)
772 end
773
774 # unary operator (i.e. "!z")
775 elseif isa(func,Symbol) && func in uni_ops && length(func_args) == 1
776 show_unquoted(io, func, indent)
777 if isa(func_args[1], Expr) || func_args[1] in all_ops
778 show_enclosed_list(io, '(', func_args, ", ", ')', indent, func_prec)
779 else
780 show_unquoted(io, func_args[1])
781 end
782
783 # binary operator (i.e. "x + y")
784 elseif func_prec > 0 # is a binary operator
785 na = length(func_args)
786 if (na == 2 || (na > 2 && func in (:+, :++, :*))) &&
787 all(!isa(a, Expr) || a.head !== :... for a in func_args)
788 sep = " $func "
789 if func_prec <= prec
790 show_enclosed_list(io, '(', func_args, sep, ')', indent, func_prec, true)
791 else
792 show_list(io, func_args, sep, indent, func_prec, true)
793 end
794 elseif na == 1
795 # 1-argument call to normally-binary operator
796 op, cl = expr_calls[head]
797 print(io, "(")
798 show_unquoted(io, func, indent)
799 print(io, ")")
800 show_enclosed_list(io, op, func_args, ", ", cl, indent)
801 else
802 show_call(io, head, func, func_args, indent)
803 end
804
805 # normal function (i.e. "f(x,y)")
806 else
807 show_call(io, head, func, func_args, indent)
808 end
809
810 # other call-like expressions ("A[1,2]", "T{X,Y}", "f.(X,Y)")
811 elseif haskey(expr_calls, head) && nargs >= 1 # :ref/:curly/:calldecl/:(.)
812 funcargslike = head == :(.) ? ex.args[2].args : ex.args[2:end]
813 show_call(io, head, ex.args[1], funcargslike, indent)
814
815 # comprehensions
816 elseif head === :typed_comprehension && length(args) == 2
817 show_unquoted(io, args[1], indent)
818 print(io, '[')
819 show_generator(io, args[2], indent)
820 print(io, ']')
821
822 elseif head === :comprehension && length(args) == 1
823 print(io, '[')
824 show_generator(io, args[1], indent)
825 print(io, ']')
826
827 elseif (head === :generator && length(args) >= 2) || (head === :flatten && length(args) == 1)
828 print(io, '(')
829 show_generator(io, ex, indent)
830 print(io, ')')
831
832 elseif head === :filter && length(args) == 2
833 show_unquoted(io, args[2], indent)
834 print(io, " if ")
835 show_unquoted(io, args[1], indent)
836
837 # comparison (i.e. "x < y < z")
838 elseif head === :comparison && nargs >= 3 && (nargs&1==1)
839 comp_prec = minimum(operator_precedence, args[2:2:end])
840 if comp_prec <= prec
841 show_enclosed_list(io, '(', args, " ", ')', indent, comp_prec)
842 else
843 show_list(io, args, " ", indent, comp_prec)
844 end
845
846 # function calls need to transform the function from :call to :calldecl
847 # so that operators are printed correctly
848 elseif head === :function && nargs==2 && is_expr(args[1], :call)
849 show_block(io, head, Expr(:calldecl, args[1].args...), args[2], indent)
850 print(io, "end")
851
852 elseif head === :function && nargs == 1
853 print(io, "function ", args[1], " end")
854
855 # block with argument
856 elseif head in (:for,:while,:function,:if) && nargs==2
857 show_block(io, head, args[1], args[2], indent)
858 print(io, "end")
859
860 elseif head === :module && nargs==3 && isa(args[1],Bool)
861 show_block(io, args[1] ? :module : :baremodule, args[2], args[3], indent)
862 print(io, "end")
863
864 # type declaration
865 elseif head === :type && nargs==3
866 show_block(io, args[1] ? Symbol("mutable struct") : Symbol("struct"), args[2], args[3], indent)
867 print(io, "end")
868
869 elseif head === :bitstype && nargs == 2
870 print(io, "primitive type ")
871 show_list(io, reverse(args), ' ', indent)
872 print(io, " end")
873
874 elseif head === :abstract && nargs == 1
875 print(io, "abstract type ")
876 show_list(io, args, ' ', indent)
877 print(io, " end")
878
879 # empty return (i.e. "function f() return end")
880 elseif head === :return && nargs == 1 && args[1] === nothing
881 print(io, head)
882
883 # type annotation (i.e. "::Int")
884 elseif head === Symbol("::") && nargs == 1
885 print(io, "::")
886 show_unquoted(io, args[1], indent)
887
888 # var-arg declaration or expansion
889 # (i.e. "function f(L...) end" or "f(B...)")
890 elseif head === :(...) && nargs == 1
891 show_unquoted(io, args[1], indent)
892 print(io, "...")
893
894 elseif (nargs == 0 && head in (:break, :continue))
895 print(io, head)
896
897 elseif (nargs == 1 && head in (:return, :const)) ||
898 head in (:local, :global, :export)
899 print(io, head, ' ')
900 show_list(io, args, ", ", indent)
901
902 elseif head === :macrocall && nargs >= 1
903 # Use the functional syntax unless specifically designated with prec=-1
904 if prec >= 0
905 show_call(io, :call, ex.args[1], ex.args[2:end], indent)
906 else
907 show_list(io, args, ' ', indent)
908 end
909
910 elseif head === :line && 1 <= nargs <= 2
911 show_linenumber(io, args...)
912
913 elseif head === :if && nargs == 3 # if/else
914 show_block(io, "if", args[1], args[2], indent)
915 show_block(io, "else", args[3], indent)
916 print(io, "end")
917
918 elseif head === :try && 3 <= nargs <= 4
919 show_block(io, "try", args[1], indent)
920 if is_expr(args[3], :block)
921 show_block(io, "catch", args[2] === false ? Any[] : args[2], args[3], indent)
922 end
923 if nargs >= 4 && is_expr(args[4], :block)
924 show_block(io, "finally", Any[], args[4], indent)
925 end
926 print(io, "end")
927
928 elseif head === :let && nargs >= 1
929 show_block(io, "let", args[2:end], args[1], indent); print(io, "end")
930
931 elseif head === :block || head === :body
932 show_block(io, "begin", ex, indent); print(io, "end")
933
934 elseif head === :quote && nargs == 1 && isa(args[1],Symbol)
935 show_unquoted_quote_expr(io, args[1], indent, 0)
936
937 elseif head === :gotoifnot && nargs == 2
938 print(io, "unless ")
939 show_list(io, args, " goto ", indent)
940
941 elseif head === :string && nargs == 1 && isa(args[1], AbstractString)
942 show(io, args[1])
943
944 elseif head === :null
945 print(io, "nothing")

947 elseif head === :kw && length(args)==2
948 show_unquoted(io, args[1], indent+indent_width)
949 print(io, '=')
950 show_unquoted(io, args[2], indent+indent_width)
951
952 elseif head === :string
953 print(io, '"')
954 for x in args
955 if !isa(x,AbstractString)
956 print(io, "\$(")
957 if isa(x,Symbol) && !(x in quoted_syms)
958 print(io, x)
959 else

961 end
962 print(io, ")")
963 else

965 end
966 end
967 print(io, '"')
968

972 parens = (isa(a1,Expr) && a1.head !== :tuple) || (isa(a1,Symbol) && isoperator(a1))

980 show_unquoted(io, args[1])
981 else
982 print(io, "(")
983 show_unquoted(io, args[1])
984 print(io, ")")
985 end
986 print(io, head)
987
988 # `where` syntax
989 elseif head === :where && length(args) > 1
990 parens = 1 <= prec
991 parens && print(io, "(")
992 show_unquoted(io, args[1], indent, operator_precedence(:(::)))
993 print(io, " where ")
994 if nargs == 2
995 show_unquoted(io, args[2], indent, 1)
996 else
997 print(io, "{")
998 show_list(io, args[2:end], ", ", indent)
999 print(io, "}")
1000 end
1001 parens && print(io, ")")
1002
1003 elseif head === :import || head === :importall || head === :using
1004 print(io, head)
1005 first = true
1006 for a = args
1007 if first
1008 print(io, ' ')
1009 first = false
1010 else
1011 print(io, '.')
1012 end
1013 if a !== :.
1014 print(io, a)
1015 end
1016 end
1017 elseif head === :meta && length(args) >= 2 && args[1] === :push_loc
1018 print(io, "# meta: location ", join(args[2:end], " "))
1019 show_type = false
1020 elseif head === :meta && length(args) == 1 && args[1] === :pop_loc
1021 print(io, "# meta: pop location")
1022 show_type = false
1023 # print anything else as "Expr(head, args...)"
1024 else
1025 show_type = false
1026 if emphstate && ex.head !== :lambda && ex.head !== :method
1027 io = IOContext(io, :TYPEEMPHASIZE => false)
1028 emphstate = false
1029 end
1030 print(io, "\$(Expr(")
1031 show(io, ex.head)
1032 for arg in args
1033 print(io, ", ")
1034 show(io, arg)
1035 end
1036 print(io, "))")
1037 end
1038 show_type && show_expr_type(io, ex.typ, emphstate)
1039 nothing
1040 end

1335 """
1336 `alignment(X, rows, cols, cols_if_complete, cols_otherwise, sep)` returns the
1337 alignment for specified parts of array `X`, returning the (left,right) info.
1338 It will look in X's `rows`, `cols` (both lists of indices)
1339 and figure out what's needed to be fully aligned, for example looking all
1340 the way down a column and finding out the maximum size of each element.
1341 Parameter `sep::Integer` is number of spaces to put between elements.
1342 `cols_if_complete` and `cols_otherwise` indicate screen width to use.
1343 Alignment is reported as a vector of (left,right) tuples, one for each
1344 column going across the screen.
1345 """
1346 function alignment(io::IO, X::AbstractVecOrMat,
1347 rows::AbstractVector, cols::AbstractVector,
1348 cols_if_complete::Integer, cols_otherwise::Integer, sep::Integer)
1349 a = Tuple{Int, Int}[]
1350 for j in cols # need to go down each column one at a time
1351 l = r = 0
1352 for i in rows # plumb down and see what largest element sizes are
1353 if isassigned(X,i,j)
1354 aij = alignment(io, X[i,j])
1355 else
1356 aij = undef_ref_alignment
1357 end
1358 l = max(l, aij[1]) # left characters
1359 r = max(r, aij[2]) # right characters
1360 end
1361 push!(a, (l, r)) # one tuple per column of X, pruned to screen width
1362 if length(a) > 1 && sum(map(sum,a)) + sep*length(a) >= cols_if_complete
1363 pop!(a) # remove this latest tuple if we're already beyond screen width
1364 break
1365 end
1366 end
1367 if 1 < length(a) < length(indices(X,2))
1368 while sum(map(sum,a)) + sep*length(a) >= cols_otherwise
1369 pop!(a)
1370 end
1371 end
1372 return a
1373 end

1435 """
1436 print_matrix(io::IO, mat, pre, sep, post, hdots, vdots, ddots, hmod, vmod)
1437
1438 Prints a matrix with limited output size. If `io` sets `:limit` to true,
1439 then only the corners of the matrix are printed, separated with vertical,
1440 horizontal, and diagonal ellipses as appropriate.
1441 Optional arguments are string pre (printed before the matrix, e.g. an opening bracket)
1442 which will cause a corresponding same-size indent on following rows, and
1443 string post (printed at the end of the last row of the matrix).
1444 Also options to use different ellipsis characters hdots, vdots, ddots.
1445 These are repeated every hmod or vmod elements.
1446 """
1447 function print_matrix(io::IO, X::AbstractVecOrMat,
1448 pre::AbstractString = " ", # pre-matrix string
1449 sep::AbstractString = " ", # separator between elements
1450 post::AbstractString = "", # post-matrix string
1451 hdots::AbstractString = " \u2026 ",
1452 vdots::AbstractString = "\u22ee",
1453 ddots::AbstractString = " \u22f1 ",
1454 hmod::Integer = 5, vmod::Integer = 5)
1455 if !get(io, :limit, false)
1456 screenheight = screenwidth = typemax(Int)
1457 else
1458 sz = displaysize(io)
1459 screenheight, screenwidth = sz[1] - 4, sz[2]
1460 end
1461 screenwidth -= length(pre) + length(post)
1462 presp = repeat(" ", length(pre)) # indent each row to match pre string
1463 postsp = ""
1464 @assert strwidth(hdots) == strwidth(ddots)
1465 sepsize = length(sep)
1466 rowsA, colsA = indices(X,1), indices(X,2)
1467 m, n = length(rowsA), length(colsA)
1468 # To figure out alignments, only need to look at as many rows as could
1469 # fit down screen. If screen has at least as many rows as A, look at A.
1470 # If not, then we only need to look at the first and last chunks of A,
1471 # each half a screen height in size.
1472 halfheight = div(screenheight,2)
1473 if m > screenheight
1474 rowsA = [rowsA[1:halfheight]; rowsA[m-div(screenheight-1,2)+1:m]]
1475 end
1476 # Similarly for columns, only necessary to get alignments for as many
1477 # columns as could conceivably fit across the screen
1478 maxpossiblecols = div(screenwidth, 1+sepsize)
1479 if n > maxpossiblecols
1480 colsA = [colsA[1:maxpossiblecols]; colsA[(n-maxpossiblecols+1):n]]
1481 end
1482 A = alignment(io, X, rowsA, colsA, screenwidth, screenwidth, sepsize)
1483 # Nine-slicing is accomplished using print_matrix_row repeatedly
1484 if m <= screenheight # rows fit vertically on screen
1485 if n <= length(A) # rows and cols fit so just print whole matrix in one piece
1486 for i in rowsA
1487 print(io, i == first(rowsA) ? pre : presp)
1488 print_matrix_row(io, X,A,i,colsA,sep)
1489 print(io, i == last(rowsA) ? post : postsp)
1490 if i != last(rowsA); println(io); end
1491 end
1492 else # rows fit down screen but cols don't, so need horizontal ellipsis
1493 c = div(screenwidth-length(hdots)+1,2)+1 # what goes to right of ellipsis
1494 Ralign = reverse(alignment(io, X, rowsA, reverse(colsA), c, c, sepsize)) # alignments for right
1495 c = screenwidth - sum(map(sum,Ralign)) - (length(Ralign)-1)*sepsize - length(hdots)
1496 Lalign = alignment(io, X, rowsA, colsA, c, c, sepsize) # alignments for left of ellipsis
1497 for i in rowsA
1498 print(io, i == first(rowsA) ? pre : presp)
1499 print_matrix_row(io, X,Lalign,i,colsA[1:length(Lalign)],sep)
1500 print(io, (i - first(rowsA)) % hmod == 0 ? hdots : repeat(" ", length(hdots)))
1501 print_matrix_row(io, X,Ralign,i,n-length(Ralign)+colsA,sep)
1502 print(io, i == last(rowsA) ? post : postsp)
1503 if i != last(rowsA); println(io); end
1504 end
1505 end
1506 else # rows don't fit so will need vertical ellipsis
1507 if n <= length(A) # rows don't fit, cols do, so only vertical ellipsis
1508 for i in rowsA
1509 print(io, i == first(rowsA) ? pre : presp)
1510 print_matrix_row(io, X,A,i,colsA,sep)
1511 print(io, i == last(rowsA) ? post : postsp)
1512 if i != rowsA[end]; println(io); end
1513 if i == rowsA[halfheight]
1514 print(io, i == first(rowsA) ? pre : presp)
1515 print_matrix_vdots(io, vdots,A,sep,vmod,1)
1516 println(io, i == last(rowsA) ? post : postsp)
1517 end
1518 end
1519 else # neither rows nor cols fit, so use all 3 kinds of dots
1520 c = div(screenwidth-length(hdots)+1,2)+1
1521 Ralign = reverse(alignment(io, X, rowsA, reverse(colsA), c, c, sepsize))
1522 c = screenwidth - sum(map(sum,Ralign)) - (length(Ralign)-1)*sepsize - length(hdots)
1523 Lalign = alignment(io, X, rowsA, colsA, c, c, sepsize)
1524 r = mod((length(Ralign)-n+1),vmod) # where to put dots on right half
1525 for i in rowsA
1526 print(io, i == first(rowsA) ? pre : presp)
1527 print_matrix_row(io, X,Lalign,i,colsA[1:length(Lalign)],sep)
1528 print(io, (i - first(rowsA)) % hmod == 0 ? hdots : repeat(" ", length(hdots)))
1529 print_matrix_row(io, X,Ralign,i,n-length(Ralign)+colsA,sep)
1530 print(io, i == last(rowsA) ? post : postsp)
1531 if i != rowsA[end]; println(io); end
1532 if i == rowsA[halfheight]
1533 print(io, i == first(rowsA) ? pre : presp)
1534 print_matrix_vdots(io, vdots,Lalign,sep,vmod,1)
1535 print(io, ddots)
1536 print_matrix_vdots(io, vdots,Ralign,sep,vmod,r)
1537 println(io, i == last(rowsA) ? post : postsp)
1538 end
1539 end
1540 end
1541 end
