198 dopushdisplay && pushdisplay(d)
199 repl_channel, response_channel = backend.repl_channel, backend.response_channel
200 hit_eof = false
201 while true
202 Base.reseteof(repl.terminal)
203 write(repl.terminal, JULIA_PROMPT)
204 line = ""
205 ast = nothing
206 interrupted = false
207 while true
208 try
209 line *= readline(repl.terminal, chomp=false)
210 catch e
211 if isa(e,InterruptException)
212 try # raise the debugger if present
213 ccall(:jl_raise_debugger, Int, ())
214 end
215 line = ""
216 interrupted = true
217 break
218 elseif isa(e,EOFError)
219 hit_eof = true
220 break
221 else
222 rethrow()
223 end
224 end
225 ast = Base.parse_input_line(line)
226 (isa(ast,Expr) && ast.head == :incomplete) || break
227 end
228 if !isempty(line)
229 put!(repl_channel, (ast, 1))
230 val, bt = take!(response_channel)
231 if !ends_with_semicolon(line)
232 print_response(repl, val, bt, true, false)
233 end
234 end
235 write(repl.terminal, '\n')

344 function hist_from_file(hp, file, path)
345 hp.history_file = file
346 seek(file, 0)
347 countlines = 0
348 while true
349 mode = :julia
350 line = hist_getline(file)
351 isempty(line) && break
352 countlines += 1
353 line[1] != '#' &&
354 error(invalid_history_message(path), repr(line[1]), " at line ", countlines)
355 while !isempty(line)
356 m = match(r"^#\s*(\w+)\s*:\s*(.*?)\s*$", line)
357 m === nothing && break
358 if m.captures[1] == "mode"
359 mode = Symbol(m.captures[2])
360 end
361 line = hist_getline(file)
362 countlines += 1
363 end
364 isempty(line) && break
365 # Make sure starts with tab
366 line[1] == ' ' &&
367 error(munged_history_message(path), countlines)
368 line[1] != '\t' &&
369 error(invalid_history_message(path), repr(line[1]), " at line ", countlines)
370 lines = String[]
371 while !isempty(line)
372 push!(lines, chomp(line[2:end]))
373 eof(file) && break
374 ch = Char(Base.peek(file))
375 ch == ' ' && error(munged_history_message(path), countlines)
376 ch != '\t' && break
377 line = hist_getline(file)
378 countlines += 1
379 end
380 push!(hp.modes, mode)
381 push!(hp.history, join(lines, '\n'))
382 end
383 seekend(file)
384 hp.start_idx = length(hp.history)
385 hp
386 end

416 function history_move(s::Union{LineEdit.MIState,LineEdit.PrefixSearchState}, hist::REPLHistoryProvider, idx::Int, save_idx::Int = hist.cur_idx)
417 max_idx = length(hist.history) + 1
418 @assert 1 <= hist.cur_idx <= max_idx
419 (1 <= idx <= max_idx) || return :none
420 idx != hist.cur_idx || return :none
421
422 # save the current line
423 if save_idx == max_idx
424 hist.last_mode = LineEdit.mode(s)
425 hist.last_buffer = copy(LineEdit.buffer(s))
426 else
427 hist.history[save_idx] = LineEdit.input_string(s)
428 hist.modes[save_idx] = mode_idx(hist, LineEdit.mode(s))
429 end
430
431 # load the saved line
432 if idx == max_idx
433 last_buffer = hist.last_buffer
434 LineEdit.transition(s, hist.last_mode) do
435 LineEdit.replace_line(s, last_buffer)
436 end
437 hist.last_mode = nothing
438 hist.last_buffer = IOBuffer()
439 else
440 if haskey(hist.mode_mapping, hist.modes[idx])
441 LineEdit.transition(s, hist.mode_mapping[hist.modes[idx]]) do
442 LineEdit.replace_line(s, hist.history[idx])
443 end
444 else
445 return :skip
446 end
447 end
448 hist.cur_idx = idx
449
450 return :ok

546 function history_search(hist::REPLHistoryProvider, query_buffer::IOBuffer, response_buffer::IOBuffer,
547 backwards::Bool=false, skip_current::Bool=false)
548
549 qpos = position(query_buffer)
550 qpos > 0 || return true
551 searchdata = beforecursor(query_buffer)
552 response_str = String(response_buffer)
553
554 # Alright, first try to see if the current match still works
555 a = position(response_buffer) + 1 # position is zero-indexed
556 b = min(endof(response_str), prevind(response_str, a + sizeof(searchdata))) # ensure that b is valid
557
558 !skip_current && searchdata == response_str[a:b] && return true
559
560 searchfunc, searchstart, skipfunc = backwards ? (rsearch, b, prevind) :
561 (search, a, nextind)
562 skip_current && (searchstart = skipfunc(response_str, searchstart))
563
564 # Start searching
565 # First the current response buffer
566 if 1 <= searchstart <= endof(response_str)
567 match = searchfunc(response_str, searchdata, searchstart)
568 if match != 0:-1
569 seek(response_buffer, first(match) - 1)
570 return true
571 end
572 end
573
574 # Now search all the other buffers
575 idxs = backwards ? ((hist.cur_idx-1):-1:1) : ((hist.cur_idx+1):length(hist.history))
576 for idx in idxs
577 h = hist.history[idx]
578 match = searchfunc(h, searchdata)
579 if match != 0:-1 && h != response_str && haskey(hist.mode_mapping, hist.modes[idx])
580 truncate(response_buffer, 0)
581 write(response_buffer, h)
582 seek(response_buffer, first(match) - 1)
583 hist.cur_idx = idx
584 return true
585 end
586 end
587
588 return false
589 end

693 function setup_interface(repl::LineEditREPL; hascolor = repl.hascolor, extra_repl_keymap = Dict{Any,Any}[])
694 ###
695 #
696 # This function returns the main interface that describes the REPL
697 # functionality, it is called internally by functions that setup a
698 # Terminal-based REPL frontend, but if you want to customize your REPL
699 # or embed the REPL in another interface, you may call this function
700 # directly and append it to your interface.
701 #
702 # Usage:
703 #
704 # repl_channel,response_channel = Channel(),Channel()
705 # start_repl_backend(repl_channel, response_channel)
706 # setup_interface(REPLDisplay(t),repl_channel,response_channel)
707 #
708 ###
709
710 ###
711 # We setup the interface in two stages.
712 # First, we set up all components (prompt,rsearch,shell,help)
713 # Second, we create keymaps with appropriate transitions between them
714 # and assign them to the components
715 #
716 ###

721 replc = REPLCompletionProvider()

724 julia_prompt = Prompt(JULIA_PROMPT;
725 # Copy colors from the prompt object
726 prompt_prefix = hascolor ? repl.prompt_color : "",
727 prompt_suffix = hascolor ?
728 (repl.envcolors ? Base.input_color : repl.input_color) : "",

796 const repl_keymap = AnyDict(
797 ';' => function (s,o...)
798 if isempty(s) || position(LineEdit.buffer(s)) == 0
799 buf = copy(LineEdit.buffer(s))
800 transition(s, shell_mode) do
801 LineEdit.state(s, shell_mode).input_buffer = buf
802 end
803 else
804 edit_insert(s, ';')
805 end
806 end,
807 '?' => function (s,o...)
808 if isempty(s) || position(LineEdit.buffer(s)) == 0
809 buf = copy(LineEdit.buffer(s))
810 transition(s, help_mode) do
811 LineEdit.state(s, help_mode).input_buffer = buf
812 end
813 else
814 edit_insert(s, '?')
815 end
816 end,
817
818 # Bracketed Paste Mode
819 "\e[200~" => (s,o...)->begin
820 input = LineEdit.bracketed_paste(s) # read directly from s until reaching the end-bracketed-paste marker
821 sbuffer = LineEdit.buffer(s)
822 curspos = position(sbuffer)
823 seek(sbuffer, 0)
824 shouldeval = (nb_available(sbuffer) == curspos && search(sbuffer, UInt8('\n')) == 0)
825 seek(sbuffer, curspos)
826 if curspos == 0
827 # if pasting at the beginning, strip leading whitespace
828 input = lstrip(input)
829 end
830 if !shouldeval
831 # when pasting in the middle of input, just paste in place
832 # don't try to execute all the WIP, since that's rather confusing
833 # and is often ill-defined how it should behave
834 edit_insert(s, input)
835 return
836 end
837 edit_insert(sbuffer, input)
838 input = String(take!(sbuffer))
839 oldpos = start(input)
840 firstline = true
841 isprompt_paste = false
842 while !done(input, oldpos) # loop until all lines have been executed
843 if JL_PROMPT_PASTE[]
844 # Check if the next statement starts with "julia> ", in that case
845 # skip it. But first skip whitespace
846 while input[oldpos] in ('\n', ' ', '\t')
847 oldpos = nextind(input, oldpos)
848 oldpos >= sizeof(input) && return
849 end
850 # Check if input line starts with "julia> ", remove it if we are in prompt paste mode
851 jl_prompt_len = 7
852 if (firstline || isprompt_paste) && (oldpos + jl_prompt_len <= sizeof(input) && input[oldpos:oldpos+jl_prompt_len-1] == JULIA_PROMPT)
853 isprompt_paste = true
854 oldpos += jl_prompt_len
855 # If we are prompt pasting and current statement does not begin with julia> , skip to next line
856 elseif isprompt_paste
857 while input[oldpos] != '\n'
858 oldpos = nextind(input, oldpos)
859 oldpos >= sizeof(input) && return
860 end
861 continue
862 end
863 end
864 ast, pos = Base.syntax_deprecation_warnings(false) do
865 Base.parse(input, oldpos, raise=false)
866 end
867 if (isa(ast, Expr) && (ast.head == :error || ast.head == :continue || ast.head == :incomplete)) ||
868 (done(input, pos) && !endswith(input, '\n'))
869 # remaining text is incomplete (an error, or parser ran to the end but didn't stop with a newline):
870 # Insert all the remaining text as one line (might be empty)
871 tail = input[oldpos:end]
872 if !firstline
873 # strip leading whitespace, but only if it was the result of executing something
874 # (avoids modifying the user's current leading wip line)
875 tail = lstrip(tail)
876 end
877 LineEdit.replace_line(s, tail)
878 LineEdit.refresh_line(s)
879 break
880 end
881 # get the line and strip leading and trailing whitespace
882 line = strip(input[oldpos:prevind(input, pos)])
883 if !isempty(line)
884 # put the line on the screen and history
885 LineEdit.replace_line(s, line)
886 LineEdit.commit_line(s)
887 # execute the statement
888 terminal = LineEdit.terminal(s) # This is slightly ugly but ok for now
889 raw!(terminal, false) && disable_bracketed_paste(terminal)
890 LineEdit.mode(s).on_done(s, LineEdit.buffer(s), true)
891 raw!(terminal, true) && enable_bracketed_paste(terminal)
892 end
893 oldpos = pos
894 firstline = false
895 end
896 end,
897
898 # Open the editor at the location of a stackframe
899 # This is accessing a global variable that gets set in
900 # the show_backtrace function.
901 "^Q" => (s, o...) -> begin
902 linfos = Base.LAST_BACKTRACE_LINE_INFOS
903 str = String(take!(LineEdit.buffer(s)))
904 n = tryparse(Int, str)
905 isnull(n) && @goto writeback
906 n = get(n)
907 if n <= 0 || n > length(linfos) || startswith(linfos[n][1], "./REPL")
908 @goto writeback
909 end
910 Base.edit(linfos[n][1], linfos[n][2])
911 Base.LineEdit.refresh_line(s)
912 return
913 @label writeback
914 write(Base.LineEdit.buffer(s), str)
915 return
916 end,
917 )
918
919 prefix_prompt, prefix_keymap = LineEdit.setup_prefix_keymap(hp, julia_prompt)
920
921 a = Dict{Any,Any}[skeymap, repl_keymap, prefix_keymap, LineEdit.history_keymap, LineEdit.default_keymap, LineEdit.escape_defaults]
922 prepend!(a, extra_repl_keymap)
923
924 julia_prompt.keymap_dict = LineEdit.keymap(a)
925
926 mk = mode_keymap(julia_prompt)
927
928 b = Dict{Any,Any}[skeymap, mk, prefix_keymap, LineEdit.history_keymap, LineEdit.default_keymap, LineEdit.escape_defaults]
929 prepend!(b, extra_repl_keymap)
930
931 shell_mode.keymap_dict = help_mode.keymap_dict = LineEdit.keymap(b)

933 ModalInterface([julia_prompt, shell_mode, help_mode, search_prompt, prefix_prompt])

977 # heuristic function to decide if the presence of a semicolon
978 # at the end of the expression was intended for suppressing output
979 function ends_with_semicolon(line)
980 match = rsearch(line, ';')
981 if match != 0
982 # state for comment parser, assuming that the `;` isn't in a string or comment
983 # so input like ";#" will still thwart this to give the wrong (anti-conservative) answer
984 comment = false
985 comment_start = false
986 comment_close = false
987 comment_multi = 0
988 for c in line[(match + 1):end]
989 if comment_multi > 0
990 # handle nested multi-line comments
991 if comment_close && c == '#'
992 comment_close = false
993 comment_multi -= 1
994 elseif comment_start && c == '='
995 comment_start = false
996 comment_multi += 1
997 else
998 comment_start = (c == '#')
999 comment_close = (c == '=')
1000 end
1001 elseif comment
1002 # handle line comments
1003 if c == '\r' || c == '\n'
1004 comment = false
1005 end
1006 elseif comment_start
1007 # see what kind of comment this is
1008 comment_start = false
1009 if c == '='
1010 comment_multi = 1
1011 else
1012 comment = true
1013 end
1014 elseif c == '#'
1015 # start handling for a comment
1016 comment_start = true
1017 else
1018 # outside of a comment, encountering anything but whitespace
1019 # means the semi-colon was internal to the expression
1020 isspace(c) || return false
1021 end
1022 end
1023 return true
1024 end
1025 return false
1026 end

1028 function run_frontend(repl::StreamREPL, backend::REPLBackendRef)
1029 have_color = Base.have_color
1030 banner(repl.stream, have_color)
1031 d = REPLDisplay(repl)
1032 dopushdisplay = !in(d,Base.Multimedia.displays)
1033 dopushdisplay && pushdisplay(d)
1034 repl_channel, response_channel = backend.repl_channel, backend.response_channel
1035 while !eof(repl.stream)
1036 if have_color
1037 print(repl.stream,repl.prompt_color)
1038 end
1039 print(repl.stream, "julia> ")
1040 if have_color
1041 print(repl.stream, input_color(repl))
1042 end
1043 line = readline(repl.stream, chomp=false)
1044 if !isempty(line)
1045 ast = Base.parse_input_line(line)
1046 if have_color
1047 print(repl.stream, Base.color_normal)
1048 end
1049 put!(repl_channel, (ast, 1))
1050 val, bt = take!(response_channel)
1051 if !ends_with_semicolon(line)
1052 print_response(repl, val, bt, true, have_color)
1053 end
1054 end
1055 end
1056 # Terminate Backend
1057 put!(repl_channel, (nothing, -1))
1058 dopushdisplay && popdisplay(d)
1059 end
