5 """
6 Condition()
7
8 Create an edge-triggered event source that tasks can wait for. Tasks that call [`wait`](@ref) on a
9 `Condition` are suspended and queued. Tasks are woken up when [`notify`](@ref) is later called on
10 the `Condition`. Edge triggering means that only tasks waiting at the time [`notify`](@ref) is
11 called can be woken up. For level-triggered notifications, you must keep extra state to keep
12 track of whether a notification has happened. The [`Channel`](@ref) type does
13 this, and so can be used for level-triggered events.
14 """

34 """
35 notify(condition, val=nothing; all=true, error=false)
36
37 Wake up tasks waiting for a condition, passing them `val`. If `all` is `true` (the default),
38 all waiting tasks are woken, otherwise only one is. If `error` is `true`, the passed value
39 is raised as an exception in the woken tasks.
40
41 Returns the count of tasks woken up. Returns 0 if no tasks are waiting on `condition`.
42 """
43 notify(c::Condition, arg::ANY=nothing; all=true, error=false) = notify(c, arg, all, error)
44 function notify(c::Condition, arg, all, error)
45 cnt = 0
46 if all
47 cnt = length(c.waitq)
48 for t in c.waitq
49 error ? schedule(t, arg, error=error) : schedule(t, arg)
50 end
51 empty!(c.waitq)
52 elseif !isempty(c.waitq)
53 cnt = 1
54 t = shift!(c.waitq)
55 error ? schedule(t, arg, error=error) : schedule(t, arg)
56 end
57 cnt
58 end

91 """
92 schedule(t::Task, [val]; error=false)
93
94 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
95 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
96
97 If a second argument `val` is provided, it will be passed to the task (via the return value of
98 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
99 the woken task.
100
101 ```jldoctest
102 julia> a5() = det(rand(1000, 1000));
103
104 julia> b = Task(a5);
105
106 julia> istaskstarted(b)
107 false
108
109 julia> schedule(b);
110
111 julia> yield();
112
113 julia> istaskstarted(b)
114 true
115
116 julia> istaskdone(b)
117 true
118 ```
119 """
120 function schedule(t::Task, arg; error=false)
121 # schedule a task to be (re)started with the given value or exception
122 if error
123 t.exception = arg
124 else
125 t.result = arg
126 end
127 return enq_work(t)
128 end

165 """
166 yieldto(t::Task, arg = nothing)
167
168 Switch to the given task. The first time a task is switched to, the task's function is
169 called with no arguments. On subsequent switches, `arg` is returned from the task's last
170 call to `yieldto`. This is a low-level call that only switches tasks, not considering states
171 or scheduling in any way. Its use is discouraged.
172 """

176 end

202 function ensure_self_descheduled()
203 # return a queued task to the runnable state
204 ct = current_task()
205 if ct.state == :queued
206 i = findfirst(Workqueue, ct)
207 i == 0 || deleteat!(Workqueue, i)
208 ct.state = :runnable

213 function wait()
214 while true
215 if isempty(Workqueue)
216 c = process_events(true)
217 if c==0 && eventloop()!=C_NULL && isempty(Workqueue)
218 # if there are no active handles and no runnable tasks, just
219 # wait for signals.
220 pause()
221 end
222 else
223 let t = shift!(Workqueue)
224 if t.state != :queued
225 # assume this somehow got queued twice,
226 # probably broken now, but try discarding this switch and keep going
227 # can't throw here, because it's probably not the fault of the caller to wait
228 # and don't want to use print() here, because that may try to incur a task switch
229 ccall(:jl_safe_printf, Void, (Ptr{UInt8}, Vararg{Int32}),
230 "\nWARNING: Workqueue inconsistency detected: shift!(Workqueue).state != :queued\n")
231 continue
232 end
233 t.state = :runnable
234 result = try_yieldto(t) do
235 # we failed to yield to t
236 # return it to the head of the queue to be scheduled later
237 unshift!(Workqueue, t)
238 t.state = :queued
239 ensure_self_descheduled()
240 end
241 process_events(false)
242 # return when we come out of the queue
243 return result
244 end
245 end
246 end
247 # unreachable
248 end

251 pause() = ccall(:Sleep, stdcall, Void, (UInt32,), 0xffffffff)
252 else
253 pause() = ccall(:pause, Void, ())
254 end
