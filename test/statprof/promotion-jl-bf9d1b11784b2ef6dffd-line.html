8 function typejoin(a::ANY, b::ANY)
9 @_pure_meta
10 if a <: b
11 return b
12 elseif b <: a
13 return a
14 elseif isa(a,UnionAll)
15 return UnionAll(a.var, typejoin(a.body, b))
16 elseif isa(b,UnionAll)
17 return UnionAll(b.var, typejoin(a, b.body))
18 elseif isa(a,TypeVar)
19 return typejoin(a.ub, b)
20 elseif isa(b,TypeVar)
21 return typejoin(a, b.ub)
22 elseif isa(a,Union)
23 return typejoin(typejoin(a.a,a.b), b)
24 elseif isa(b,Union)
25 return typejoin(a, typejoin(b.a,b.b))
26 elseif a <: Tuple
27 if !(b <: Tuple)
28 return Any
29 end
30 ap, bp = a.parameters, b.parameters
31 lar = length(ap)::Int; lbr = length(bp)::Int
32 laf, afixed = full_va_len(ap)
33 lbf, bfixed = full_va_len(bp)
34 if lar==0 || lbr==0
35 return Tuple
36 end
37 if laf < lbf
38 if isvarargtype(ap[lar]) && !afixed
39 c = Vector{Any}(laf)
40 c[laf] = Vararg{typejoin(unwrapva(ap[lar]), tailjoin(bp,laf))}
41 n = laf-1
42 else
43 c = Vector{Any}(laf+1)
44 c[laf+1] = Vararg{tailjoin(bp,laf+1)}
45 n = laf
46 end
47 elseif lbf < laf
48 if isvarargtype(bp[lbr]) && !bfixed
49 c = Vector{Any}(lbf)
50 c[lbf] = Vararg{typejoin(unwrapva(bp[lbr]), tailjoin(ap,lbf))}
51 n = lbf-1
52 else
53 c = Vector{Any}(lbf+1)
54 c[lbf+1] = Vararg{tailjoin(ap,lbf+1)}
55 n = lbf
56 end
57 else
58 c = Vector{Any}(laf)
59 n = laf
60 end
61 for i = 1:n
62 ai = ap[min(i,lar)]; bi = bp[min(i,lbr)]
63 ci = typejoin(unwrapva(ai),unwrapva(bi))
64 c[i] = i == length(c) && (isvarargtype(ai) || isvarargtype(bi)) ? Vararg{ci} : ci
65 end
66 return Tuple{c...}
67 elseif b <: Tuple
68 return Any
69 end
70 while b !== Any
71 if a <: b.name.wrapper
72 while a.name !== b.name
73 a = supertype(a)
74 end
75 aprimary = unwrap_unionall(a.name.wrapper)
76 # join on parameters
77 n = length(a.parameters)
78 if n == 0
79 return aprimary
80 end
81 p = Vector{Any}(n)
82 for i = 1:n
83 ai, bi = a.parameters[i], b.parameters[i]
84 if ai === bi || (isa(ai,Type) && isa(bi,Type) && typeseq(ai,bi))
85 p[i] = ai
86 else
87 p[i] = aprimary.parameters[i]
88 end
89 end
90 return rewrap_unionall(a.name.wrapper{p...}, a.name.wrapper)
91 end
92 b = supertype(b)
93 end
94 return Any
95 end

98 function full_va_len(p)
99 isempty(p) && return 0, true
100 last = p[end]
101 if isvarargtype(last)
102 N = unwrap_unionall(last).parameters[2]
103 if isa(N, Integer)
104 return (length(p) + N - 1)::Int, true
105 end
106 return length(p)::Int, false
107 end
108 return length(p)::Int, true
109 end

120 return t

134 """
135 promote_type(type1, type2)
136
137 Determine a type big enough to hold values of each argument type without loss, whenever
138 possible. In some cases, where no type exists to which both types can be promoted
139 losslessly, some loss is tolerated; for example, `promote_type(Int64, Float64)` returns
140 [`Float64`](@ref) even though strictly, not all [`Int64`](@ref) values can be represented
141 exactly as `Float64` values.
142
143 ```jldoctest
144 julia> promote_type(Int64, Float64)
145 Float64
146
147 julia> promote_type(Int32, Int64)

149
150 julia> promote_type(Float32, BigInt)

152 ```

254 """
255 ^(x, y)
256
257 Exponentiation operator. If `x` is a matrix, computes matrix exponentiation.
258
259 If `y` is an `Int` literal (e.g. `2` in `x^2` or `-3` in `x^-3`), the Julia code
260 `x^y` is transformed by the compiler to `Base.literal_pow(^, x, Val{y})`, to
261 enable compile-time specialization on the value of the exponent.
262 (As a default fallback we have `Base.literal_pow(^, x, Val{y}) = ^(x,y)`,
263 where usually `^ == Base.^` unless `^` has been defined in the calling
264 namespace.)
