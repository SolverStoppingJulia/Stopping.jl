79 function MathProgNLPModel(mpmodel :: MathProgModel; name :: String="Generic")
80
81 nvar = mpmodel.numVar

83 uvar = mpmodel.uvar
84
85 nlin = length(mpmodel.eval.m.linconstr) # Number of linear constraints.
86 nquad = length(mpmodel.eval.m.quadconstr) # Number of quadratic constraints.
87 nnln = length(mpmodel.eval.m.nlpdata.nlconstr) # Number of nonlinear constraints.
88 ncon = mpmodel.numConstr # Total number of constraints.
89 lcon = mpmodel.lcon
90 ucon = mpmodel.ucon
91
92 jrows, jcols = MathProgBase.jac_structure(mpmodel.eval)
93 hrows, hcols = MathProgBase.hesslag_structure(mpmodel.eval)
94 nnzj = length(jrows)
95 nnzh = length(hrows)
96
97 meta = NLPModelMeta(nvar,
98 x0=mpmodel.x,
99 lvar=lvar,
100 uvar=uvar,
101 ncon=ncon,
102 y0=zeros(ncon),
103 lcon=lcon,
104 ucon=ucon,
105 nnzj=nnzj,
106 nnzh=nnzh,
107 lin=collect(1:nlin), # linear constraints appear first in MPB
108 nln=collect(nlin+1:ncon),
109 minimize=(mpmodel.sense == :Min),
110 islp=MathProgBase.isobjlinear(mpmodel.eval) & (nlin == ncon),
111 name=name,
112 )
113
114 return MathProgNLPModel(meta,
115 mpmodel,
116 Counters(),
117 jrows,
118 jcols,
