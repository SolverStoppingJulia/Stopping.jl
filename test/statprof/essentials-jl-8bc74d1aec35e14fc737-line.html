99 function rename_unionall(u::ANY)
100 if !isa(u,UnionAll)
101 return u
102 end
103 body = rename_unionall(u.body)
104 if body === u.body
105 body = u
106 else
107 body = UnionAll(u.var, body)

209 """
210 @inbounds(blk)
211
212 Eliminates array bounds checking within expressions.
213
214 In the example below the bound check of array A is skipped to improve performance.
215
216 ```julia
217 function sum(A::AbstractArray)
218 r = zero(eltype(A))
219 for i = 1:length(A)
220 @inbounds r += A[i]
221 end
222 return r
223 end
224 ```
225
226 !!! warning
227
228 Using `@inbounds` may return incorrect results/crashes/corruption
229 for out-of-bounds indices. The user is responsible for checking it manually.
230 """
231 macro inbounds(blk)
232 :($(Expr(:inbounds,true));
233 $(esc(blk));
234 $(Expr(:inbounds,:pop)))
235 end

360 """
361 invokelatest(f, args...)
362
363 Calls `f(args...)`, but guarantees that the most recent method of `f`
364 will be executed. This is useful in specialized circumstances,
365 e.g. long-running event loops or callback functions that may
366 call obsolete versions of a function `f`.
367 (The drawback is that `invokelatest` is somewhat slower than calling
368 `f` directly, and the type of the result cannot be inferred by the compiler.)
369 """
370 invokelatest(f, args...) = Core._apply_latest(f, args)
