40 """
41 fullname(m::Module)
42
43 Get the fully-qualified name of a module as a tuple of symbols. For example,
44
45 ```jldoctest
46 julia> fullname(Base.Pkg)
47 (:Base, :Pkg)
48
49 julia> fullname(Main)
50 ()
51 ```
52 """
53 function fullname(m::Module)
54 m === Main && return ()
55 m === Base && return (:Base,) # issue #10653
56 mn = module_name(m)
57 mp = module_parent(m)
58 if mp === m
59 # not Main, but is its own parent, means a prior Main module
60 n = ()
61 this = Main
62 while this !== m
63 if isdefined(this, :LastMain)
64 n = tuple(n..., :LastMain)
65 this = this.LastMain
66 else
67 error("no reference to module ", mn)
68 end
69 end
70 return n
71 end
72 return tuple(fullname(mp)..., mn)
73 end

610 e = mc.targ::Vector{Any}

683 # Printing code representations in IR and assembly
684 function _dump_function(f::ANY, t::ANY, native::Bool, wrapper::Bool,
685 strip_ir_metadata::Bool, dump_module::Bool, syntax::Symbol=:att,
686 optimize::Bool=true, params::CodegenParams=CodegenParams())
687 ccall(:jl_is_in_pure_context, Bool, ()) && error("code reflection cannot be used from generated functions")
688 if isa(f, Core.Builtin)
689 throw(ArgumentError("argument is not a generic function"))
690 end
691 # get the MethodInstance for the method match
692 world = typemax(UInt)
693 meth = which(f, t)
694 t = to_tuple_type(t)
695 ft = isa(f, Type) ? Type{f} : typeof(f)
696 tt = Tuple{ft, t.parameters...}
697 (ti, env) = ccall(:jl_match_method, Any, (Any, Any), tt, meth.sig)::SimpleVector
698 meth = func_for_method_checked(meth, tt)
699 linfo = ccall(:jl_specializations_get_linfo, Ref{Core.MethodInstance}, (Any, Any, Any, UInt), meth, tt, env, world)
700 # get the code for it
701 return _dump_function_linfo(linfo, world, native, wrapper, strip_ir_metadata, dump_module, syntax, optimize, params)
702 end
703
704 function _dump_function_linfo(linfo::Core.MethodInstance, world::UInt, native::Bool, wrapper::Bool,
705 strip_ir_metadata::Bool, dump_module::Bool, syntax::Symbol=:att,
706 optimize::Bool=true, params::CodegenParams=CodegenParams())
707 if syntax != :att && syntax != :intel
708 throw(ArgumentError("'syntax' must be either :intel or :att"))
709 end
710 if native
711 llvmf = ccall(:jl_get_llvmf_decl, Ptr{Void}, (Any, UInt, Bool, CodegenParams), linfo, world, wrapper, params)
712 else
713 llvmf = ccall(:jl_get_llvmf_defn, Ptr{Void}, (Any, UInt, Bool, Bool, CodegenParams), linfo, world, wrapper, optimize, params)
714 end
715 if llvmf == C_NULL
716 error("could not compile the specified method")
717 end
718
719 if native
720 str = ccall(:jl_dump_function_asm, Ref{String},
721 (Ptr{Void}, Cint, Ptr{UInt8}), llvmf, 0, syntax)
722 else
723 str = ccall(:jl_dump_function_ir, Ref{String},
724 (Ptr{Void}, Bool, Bool), llvmf, strip_ir_metadata, dump_module)
725 end
726
727 # TODO: use jl_is_cacheable_sig instead of isleaftype
728 isleaftype(linfo.specTypes) || (str = "; WARNING: This code may not match what actually runs.\n" * str)
729 return str
730 end

773 function code_typed(f::ANY, types::ANY=Tuple; optimize=true)
774 ccall(:jl_is_in_pure_context, Bool, ()) && error("code reflection cannot be used from generated functions")

809 """
810 which(f, types)
811
812 Returns the method of `f` (a `Method` object) that would be called for arguments of the given `types`.
813
814 If `types` is an abstract type, then the method that would be called by `invoke` is returned.
815 """
816 function which(f::ANY, t::ANY)
817 if isa(f, Core.Builtin)
818 throw(ArgumentError("argument is not a generic function"))
819 end
820 t = to_tuple_type(t)
821 if isleaftype(t)
822 ms = methods(f, t)
823 isempty(ms) && error("no method found for the specified argument types")
824 length(ms)!=1 && error("no unique matching method for the specified argument types")
825 return first(ms)
826 else
827 ft = isa(f,Type) ? Type{f} : typeof(f)
828 tt = Tuple{ft, t.parameters...}
829 m = ccall(:jl_gf_invoke_lookup, Any, (Any, UInt), tt, typemax(UInt))
830 if m === nothing
831 error("no method found for the specified argument types")
832 end
833 meth = m.func::Method
834 if ccall(:jl_has_call_ambiguities, Cint, (Any, Any), tt, meth) != 0
835 error("method match is ambiguous for the specified argument types")
836 end
837 return meth
838 end
839 end

938 """
939 isambiguous(m1, m2; ambiguous_bottom=false) -> Bool
940
941 Determine whether two methods `m1` and `m2` (typically of the same
942 function) are ambiguous. This test is performed in the context of
943 other methods of the same function; in isolation, `m1` and `m2` might
944 be ambiguous, but if a third method resolving the ambiguity has been
945 defined, this returns `false`.
946
947 For parametric types, the `ambiguous_bottom` keyword argument controls whether
948 `Union{}` counts as an ambiguous intersection of type parameters √¢¬Ä¬ì√Ç¬†when `true`,
949 it is considered ambiguous, when `false` it is not. For example:
950
951 ```jldoctest
952 julia> foo(x::Complex{<:Integer}) = 1
953 foo (generic function with 1 method)
954
955 julia> foo(x::Complex{<:Rational}) = 2
956 foo (generic function with 2 methods)
957
958 julia> m1, m2 = collect(methods(foo));
959
960 julia> typeintersect(m1.sig, m2.sig)
961 Tuple{#foo,Complex{Union{}}}
962
963 julia> Base.isambiguous(m1, m2, ambiguous_bottom=true)
964 true
965
966 julia> Base.isambiguous(m1, m2, ambiguous_bottom=false)
967 false
968 ```
969 """
970 function isambiguous(m1::Method, m2::Method; ambiguous_bottom::Bool=false)
971 ti = typeintersect(m1.sig, m2.sig)
972 ti === Bottom && return false
973 if !ambiguous_bottom
974 has_bottom_parameter(ti) && return false
975 end
976 ml = _methods_by_ftype(ti, -1, typemax(UInt))
977 isempty(ml) && return true
978 for m in ml
979 if ti <: m[3].sig
980 return false
981 end
982 end
983 return true
984 end
985
986 """
987 has_bottom_parameter(t) -> Bool
988
989 Determine whether `t` is a Type for which one or more of its parameters is `Union{}`.
990 """
991 function has_bottom_parameter(t::Type)
992 ret = false
993 for p in t.parameters
994 ret |= (p == Bottom) || has_bottom_parameter(p)
995 end
996 ret
997 end
998 has_bottom_parameter(t::UnionAll) = has_bottom_parameter(unwrap_unionall(t))
999 has_bottom_parameter(t::Union) = has_bottom_parameter(t.a) & has_bottom_parameter(t.b)
1000 has_bottom_parameter(t::TypeVar) = has_bottom_parameter(t.ub)
1001 has_bottom_parameter(::Any) = false
