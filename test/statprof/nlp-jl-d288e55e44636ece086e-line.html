857 function hessian_slice(d, ex, x, H, scale, nzcount, recovery_tmp_storage,::Type{Val{CHUNK}}) where CHUNK
858
859 nzthis = length(ex.hess_I)
860 if ex.linearity == LINEAR
861 @assert nzthis == 0
862 return 0
863 end
864 R = ex.seed_matrix
865 Coloring.prepare_seed_matrix!(R,ex.rinfo)
866 local_to_global_idx = ex.rinfo.local_indices
867
868 zero_ϵ = zero(ForwardDiff.Partials{CHUNK,Float64})
869
870 input_ϵ_raw = d.input_ϵ
871 output_ϵ_raw = d.output_ϵ
872 input_ϵ = reinterpret_unsafe(ForwardDiff.Partials{CHUNK,Float64}, input_ϵ_raw)
873 output_ϵ = reinterpret_unsafe(ForwardDiff.Partials{CHUNK,Float64}, output_ϵ_raw)
874
875
876 # compute hessian-vector products
877 num_products = size(R,2) # number of hessian-vector products
878 num_chunks = div(num_products, CHUNK)
879 @assert size(R,1) == length(local_to_global_idx)
880 numVar = length(x)
881
882 for k in 1:CHUNK:CHUNK*num_chunks
883
884 for r in 1:length(local_to_global_idx)
885 # set up directional derivatives
886 @inbounds idx = local_to_global_idx[r]
887 # load up R[r,k,k+1,...,k+CHUNK-1] into input_ϵ
888 for s in 1:CHUNK
889 input_ϵ_raw[(idx-1)*CHUNK + s] = R[r,k+s-1]
890 end
891 @inbounds output_ϵ[idx] = zero_ϵ
892 end
893
894 hessian_slice_inner(d, ex, R, input_ϵ, output_ϵ, Val{CHUNK})
895
896 # collect directional derivatives
897 for r in 1:length(local_to_global_idx)
898 idx = local_to_global_idx[r]
899 # load output_ϵ into R[r,k,k+1,...,k+CHUNK-1]
900 for s in 1:CHUNK
901 R[r,k+s-1] = output_ϵ_raw[(idx-1)*CHUNK + s]
902 end
903 @inbounds input_ϵ[idx] = zero_ϵ
904 end
905
906 end
907
908 # leftover chunk
909 remaining = num_products - CHUNK*num_chunks
910 if remaining > 0
911 k = CHUNK*num_chunks+1
912 for r in 1:length(local_to_global_idx)
913 # set up directional derivatives
914 @inbounds idx = local_to_global_idx[r]
915 # load up R[r,k,k+1,...,k+remaining-1] into input_ϵ
916 for s in 1:remaining
917 # leave junk in the unused components
918 input_ϵ_raw[(idx-1)*CHUNK + s] = R[r,k+s-1]
919 end
920 @inbounds output_ϵ[idx] = zero_ϵ
921 end
922
923 hessian_slice_inner(d, ex, R, input_ϵ, output_ϵ, Val{CHUNK})
924
925 # collect directional derivatives
926 for r in 1:length(local_to_global_idx)
927 idx = local_to_global_idx[r]
928 # load output_ϵ into R[r,k,k+1,...,k+remaining-1]
929 for s in 1:remaining
930 R[r,k+s-1] = output_ϵ_raw[(idx-1)*CHUNK + s]
931 end
932 @inbounds input_ϵ[idx] = zero_ϵ
933 end
934 end
935
936 # Output is in R, now recover
937
938 #output_slice = view(H, (nzcount+1):(nzcount+nzthis))
939 output_slice = VectorView(nzcount, nzthis, pointer(H))
940 Coloring.recover_from_matmat!(output_slice, R, ex.rinfo, recovery_tmp_storage)
941 scale!(output_slice, scale)
942 return nzthis
943
944 end

1427 op = ex.args[1]

1430 ub = 0.0
1431 elseif op == :(<=) || op == :(≤)

1434 elseif op == :(>=) || op == :(≥)
