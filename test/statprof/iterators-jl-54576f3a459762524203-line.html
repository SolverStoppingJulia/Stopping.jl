35 """
36 enumerate(iter)
37
38 An iterator that yields `(i, x)` where `i` is a counter starting at 1,
39 and `x` is the `i`th value from the given iterator. It's useful when
40 you need not only the values `x` over which you are iterating, but
41 also the number of iterations so far. Note that `i` may not be valid
42 for indexing `iter`; it's also possible that `x != iter[i]`, if `iter`
43 has indices that do not start at 1. See the `enumerate(IndexLinear(),
44 iter)` method if you want to ensure that `i` is an index.
45
46 # Example
47
48 ```jldoctest
49 julia> a = ["a", "b", "c"];
50
51 julia> for (index, value) in enumerate(a)
52 println("\$index \$value")
53 end
54 1 a
55 2 b
56 3 c
57 ```
58 """
59 enumerate(iter) = Enumerate(iter)

75 struct IndexValue{I,A<:AbstractArray}
76 data::A
77 itr::I
78 end
79
80 """
81 enumerate(IndexLinear(), A)
82 enumerate(IndexCartesian(), A)
83 enumerate(IndexStyle(A), A)
84
85 An iterator that accesses each element of the array `A`, returning
86 `(i, x)`, where `i` is the index for the element and `x = A[i]`. This
87 is similar to `enumerate(A)`, except `i` will always be a valid index
88 for `A`.
89
90 Specifying `IndexLinear()` ensures that `i` will be an integer;
91 specifying `IndexCartesian()` ensures that `i` will be a
92 `CartesianIndex`; specifying `IndexStyle(A)` chooses whichever has
93 been defined as the native indexing style for array `A`.
94
95 # Examples
96
97 ```jldoctest
98 julia> A = ["a" "d"; "b" "e"; "c" "f"];
99
100 julia> for (index, value) in enumerate(IndexStyle(A), A)
101 println("\$index \$value")
102 end
103 1 a
104 2 b
105 3 c
106 4 d
107 5 e
108 6 f
109
110 julia> S = view(A, 1:2, :);
111
112 julia> for (index, value) in enumerate(IndexStyle(S), S)
113 println("\$index \$value")
114 end
115 CartesianIndex{2}((1, 1)) a
116 CartesianIndex{2}((2, 1)) b
117 CartesianIndex{2}((1, 2)) d
118 CartesianIndex{2}((2, 2)) e
119 ```
120
121 Note that `enumerate(A)` returns `i` as a *counter* (always starting
122 at 1), whereas `enumerate(IndexLinear(), A)` returns `i` as an *index*
123 (starting at the first linear index of `A`, which may or may not be
124 1).
125
126 See also: [`IndexStyle`](@ref), [`indices`](@ref).
127 """
128 enumerate(::IndexLinear, A::AbstractArray) = IndexValue(A, linearindices(A))
129 enumerate(::IndexCartesian, A::AbstractArray) = IndexValue(A, CartesianRange(indices(A)))

251 struct Filter{F,I}
252 flt::F
253 itr::I
254 end
255
256 """
257 Iterators.filter(flt, itr)
258
259 Given a predicate function `flt` and an iterable object `itr`, return an
260 iterable object which upon iteration yields the elements `x` of `itr` that
261 satisfy `flt(x)`. The order of the original iterator is preserved.
262
263 This function is *lazy*; that is, it is guaranteed to return in ``Θ(1)`` time
264 and use ``Θ(1)`` additional space, and `flt` will not be called by an
265 invocation of `filter`. Calls to `flt` will be made when iterating over the
266 returned iterable object. These calls are not cached and repeated calls will be
267 made when reiterating.
268
269 See [`Base.filter`](@ref) for an eager implementation of filtering for arrays.
270 """
271 filter(flt, itr) = Filter(flt, itr)
272
273 start(f::Filter) = start_filter(f.flt, f.itr)
274 function start_filter(pred, itr)
275 s = start(itr)
276 while !done(itr,s)
277 v,t = next(itr,s)
278 if pred(v)
279 return (false, v, t)
280 end
281 s=t
282 end
283 (true,)
284 end
285
286 next(f::Filter, s) = advance_filter(f.flt, f.itr, s)
287 function advance_filter(pred, itr, st)
288 _, v, s = st
289 while !done(itr,s)
290 w,t = next(itr,s)
291 if pred(w)
292 return v, (false, w, t)
293 end
294 s=t
295 end
296 v, (true, v, s)
297 end
298
299 done(f::Filter, s) = s[1]
300

635 if isnull(nv2)
636 v2, s2 = next(p.b, s2)
637 else
638 v2 = nv2.value
639 end
640
641 if done(p.a, s1)
642 return (v1,v2), (start(p.a), s2, Nullable{eltype(nv2)}(), done(p.b,s2))
643 end
644 return (v1,v2), (s1, s2, Nullable(v2), false)
645 end
646
647 @inline next(p::Prod2, st) = prod_next(p, st)
648 @inline done(p::AbstractProdIterator, st) = st[4]

651 struct Prod{I1, I2<:AbstractProdIterator} <: AbstractProdIterator

653 b::I2
654 end

663 x = prod_next(p, st)
664 ((x[1][1],x[1][2]...), x[2])
665 end

678 it::I
679 end

681 """
682 flatten(iter)
683
684 Given an iterator that yields iterators, return an iterator that yields the
685 elements of those iterators.
686 Put differently, the elements of the argument iterator are concatenated.
687
688 # Example
689
690 ```jldoctest
691 julia> collect(Iterators.flatten((1:2, 8:9)))
692 4-element Array{Int64,1}:
693 1
694 2
695 8
696 9
697 ```
698 """
699 flatten(itr) = Flatten(itr)
700
701 eltype(::Type{Flatten{I}}) where {I} = eltype(eltype(I))
702 iteratorsize(::Type{Flatten{I}}) where {I} = SizeUnknown()
703 iteratoreltype(::Type{Flatten{I}}) where {I} = _flatteneltype(I, iteratoreltype(I))
704 _flatteneltype(I, ::HasEltype) = iteratoreltype(eltype(I))
705 _flatteneltype(I, et) = EltypeUnknown()
706
707 function start(f::Flatten)
708 local inner, s2
709 s = start(f.it)
710 d = done(f.it, s)
711 # this is a simple way to make this function type stable
712 d && throw(ArgumentError("argument to Flatten must contain at least one iterator"))
713 while !d
714 inner, s = next(f.it, s)
715 s2 = start(inner)
716 !done(inner, s2) && break
717 d = done(f.it, s)
718 end
719 return s, inner, s2
720 end
721
722 @inline function next(f::Flatten, state)
723 s, inner, s2 = state
724 val, s2 = next(inner, s2)
725 while done(inner, s2) && !done(f.it, s)
726 inner, s = next(f.it, s)
727 s2 = start(inner)
728 end
729 return val, (s, inner, s2)
730 end
731
732 @inline function done(f::Flatten, state)
733 s, inner, s2 = state
734 return done(f.it, s) && done(inner, s2)
735 end

738 """
739 partition(collection, n)
740
741 Iterate over a collection `n` elements at a time.
742
743 # Example
744
745 ```jldoctest
746 julia> collect(Iterators.partition([1,2,3,4,5], 2))
747 3-element Array{Array{Int64,1},1}:
748 [1, 2]
749 [3, 4]
750 [5]
751 ```
752 """
753 partition(c::T, n::Integer) where {T} = PartitionIterator{T}(c, Int(n))
754
755
756 mutable struct PartitionIterator{T}
757 c::T
758 n::Int
759 end
760
761 eltype(::Type{PartitionIterator{T}}) where {T} = Vector{eltype(T)}
762
763 function length(itr::PartitionIterator)
764 l = length(itr.c)
765 return div(l, itr.n) + ((mod(l, itr.n) > 0) ? 1 : 0)
766 end
767
768 start(itr::PartitionIterator) = start(itr.c)
769
770 done(itr::PartitionIterator, state) = done(itr.c, state)
771
772 function next(itr::PartitionIterator{<:Vector}, state)
773 l = state
774 r = min(state + itr.n-1, length(itr.c))
775 return view(itr.c, l:r), r + 1
776 end
777
778 function next(itr::PartitionIterator, state)
779 v = Vector{eltype(itr.c)}(itr.n)
780 i = 0
781 while !done(itr.c, state) && i < itr.n
782 i += 1
783 v[i], state = next(itr.c, state)
784 end
785 return resize!(v, i), state
786 end
