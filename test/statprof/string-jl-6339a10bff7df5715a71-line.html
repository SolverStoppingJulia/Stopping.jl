170 @noinline function slow_utf8_next(p::Ptr{UInt8}, b::UInt8, i::Int, l::Int)
171 if is_valid_continuation(b)
172 throw(UnicodeError(UTF_ERR_INVALID_INDEX, i, unsafe_load(p,i)))
173 end
174 trailing = utf8_trailing[b + 1]
175 if l < i + trailing
176 return '\ufffd', i+1
177 end
178 c::UInt32 = 0
179 for j = 1:(trailing + 1)
180 c <<= 6
181 c += unsafe_load(p,i)
182 i += 1
183 end
184 c -= utf8_offset[trailing + 1]
185 return Char(c), i
186 end

207 function first_utf8_byte(ch::Char)
208 c = UInt32(ch)
209 b = c < 0x80 ? c%UInt8 :

248 function search(s::String, c::Char, i::Integer = 1)
249 if i < 1 || i > sizeof(s)
250 i == sizeof(s) + 1 && return 0
251 throw(BoundsError(s, i))
252 end
253 if is_valid_continuation(codeunit(s,i))
254 throw(UnicodeError(UTF_ERR_INVALID_INDEX, i, codeunit(s,i)))
255 end
256 c < Char(0x80) && return search(s, c%UInt8, i)
257 while true
258 i = search(s, first_utf8_byte(c), i)
259 (i==0 || s[i] == c) && return i
260 i = next(s,i)[2]
261 end

370 elseif c < 0x10000

426 function repeat(s::String, r::Integer)
427 r < 0 && throw(ArgumentError("can't repeat a string $r times"))
428 n = s.len
429 out = _string_n(n*r)
430 if n == 1 # common case: repeating a single ASCII char
431 ccall(:memset, Ptr{Void}, (Ptr{UInt8}, Cint, Csize_t), out, unsafe_load(pointer(s)), r)
432 else
433 for i=1:r
434 unsafe_copy!(pointer(out, 1+(i-1)*n), pointer(s), n)
435 end
436 end
