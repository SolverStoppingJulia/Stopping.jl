37 struct GC_Diff
38 allocd ::Int64 # Bytes allocated
39 malloc ::Int64 # Number of GC aware malloc()
40 realloc ::Int64 # Number of GC aware realloc()
41 poolalloc ::Int64 # Number of pool allocation
42 bigalloc ::Int64 # Number of big (non-pool) allocation
43 freecall ::Int64 # Number of GC aware free()
44 total_time ::Int64 # Time spent in garbage collection
45 pause ::Int64 # Number of GC pauses
46 full_sweep ::Int64 # Number of GC full collection

49 gc_total_bytes(gc_num::GC_Num) =
50 (gc_num.allocd + gc_num.deferred_alloc +
51 Int64(gc_num.collect) + Int64(gc_num.total_allocd))

53 function GC_Diff(new::GC_Num, old::GC_Num)
54 # logic from `src/gc.c:jl_gc_total_bytes`
55 old_allocd = gc_total_bytes(old)
56 new_allocd = gc_total_bytes(new)
57 return GC_Diff(new_allocd - old_allocd,
58 Int64(new.malloc - old.malloc),
59 Int64(new.realloc - old.realloc),
60 Int64(new.poolalloc - old.poolalloc),
61 Int64(new.bigalloc - old.bigalloc),
62 Int64(new.freecall - old.freecall),
63 Int64(new.total_time - old.total_time),
64 new.pause - old.pause,
65 new.full_sweep - old.full_sweep)
66 end

68 function gc_alloc_count(diff::GC_Diff)
69 diff.malloc + diff.realloc + diff.poolalloc + diff.bigalloc

298 # measure bytes allocated without *most* contamination from compilation
299 # Note: This reports a different value from the @time macros, because
300 # it wraps the call in a function, however, this means that things
301 # like: @allocated y = foo()
302 # will not work correctly, because it will set y in the context of
303 # the local function made by the macro, not the current function
304 """
305 @allocated
306
307 A macro to evaluate an expression, discarding the resulting value, instead returning the
308 total number of bytes allocated during evaluation of the expression. Note: the expression is
309 evaluated inside a local function, instead of the current context, in order to eliminate the
310 effects of compilation, however, there still may be some allocations due to JIT compilation.
311 This also makes the results inconsistent with the `@time` macros, which do not try to adjust
312 for the effects of compilation.
313
314 See also [`@time`](@ref), [`@timev`](@ref), [`@timed`](@ref),
315 and [`@elapsed`](@ref).
316
317 ```julia-repl
318 julia> @allocated rand(10^6)
319 8000080
320 ```
321 """
322 macro allocated(ex)
323 quote
324 let
325 local f
326 function f()
327 b0 = gc_bytes()
328 $(esc(ex))
329 gc_bytes() - b0
330 end
331 f()
332 end
333 end

432 function _redirect(io::IO, log_to::Dict, sf::StackTraces.StackFrame)
433 isnull(sf.linfo) && return io
434 mod = get(sf.linfo).def.module
435 fun = sf.func
436 if haskey(log_to, (mod,fun))
437 return log_to[(mod,fun)]
438 elseif haskey(log_to, (mod,nothing))
439 return log_to[(mod,nothing)]
440 elseif haskey(log_to, (nothing,nothing))
441 return log_to[(nothing,nothing)]
442 else
443 return io

447 function _redirect(io::IO, log_to::Dict, fun::Symbol)
448 clos = string("#",fun,"#")
449 kw = string("kw##",fun)
450 local sf
451 break_next_frame = false
452 for trace in backtrace()
453 stack::Vector{StackFrame} = StackTraces.lookup(trace)
454 filter!(frame -> !frame.from_c, stack)
455 for frame in stack
456 isnull(frame.linfo) && continue
457 sf = frame
458 break_next_frame && (@goto skip)
459 get(frame.linfo).def.module == Base || continue
460 sff = string(frame.func)
461 if frame.func == fun || startswith(sff, clos) || startswith(sff, kw)
462 break_next_frame = true
463 end
464 end
465 end
466 @label skip
467 _redirect(io, log_to, sf)
468 end

637 `securezero!` fills the memory associated with an object `o` with zeros.

644 securezero!(s::String) = unsafe_securezero!(pointer(s), sizeof(s))
645 @noinline unsafe_securezero!{T}(p::Ptr{T}, len::Integer=1) =

649 if is_windows()
650 function getpass(prompt::AbstractString)
651 print(prompt)
652 flush(STDOUT)
653 p = Vector{UInt8}(128) # mimic Unix getpass in ignoring more than 128-char passwords
654 # (also avoids any potential memory copies arising from push!)
655 try
656 plen = 0
657 while true
658 c = ccall(:_getch, UInt8, ())
659 if c == 0xff || c == UInt8('\n') || c == UInt8('\r')
660 break # EOF or return
661 elseif c == 0x00 || c == 0xe0
662 ccall(:_getch, UInt8, ()) # ignore function/arrow keys
663 elseif c == UInt8('\b') && plen > 0
664 plen -= 1 # delete last character on backspace
665 elseif !iscntrl(Char(c)) && plen < 128
666 p[plen += 1] = c
667 end
668 end
669 return unsafe_string(pointer(p), plen) # use unsafe_string rather than String(p[1:plen])
670 # to be absolutely certain we never make an extra copy
671 finally
672 securezero!(p)
673 end
674
675 return ""
676 end
677 else
678 getpass(prompt::AbstractString) = unsafe_string(ccall(:getpass, Cstring, (Cstring,), prompt))

691 const CREDUIWIN_GENERIC = 0x0001
692 const CREDUIWIN_IN_CRED_ONLY = 0x0020
693 const CREDUIWIN_ENUMERATE_CURRENT_USER = 0x0200
694
695 const CRED_PACK_GENERIC_CREDENTIALS = 0x0004
696
697 const ERROR_SUCCESS = 0x0000
698 const ERROR_CANCELLED = 0x04c7
699
700 function winprompt(message, caption, default_username; prompt_username = true)
701 # Step 1: Create an encrypted username/password bundle that will be used to set
702 # the default username (in theory could also provide a default password)
703 credbuf = Array{UInt8,1}(1024)
704 credbufsize = Ref{UInt32}(sizeof(credbuf))
705 succeeded = ccall((:CredPackAuthenticationBufferW, "credui.dll"), stdcall, Bool,
706 (UInt32, Cwstring, Cwstring, Ptr{UInt8}, Ptr{UInt32}),
707 CRED_PACK_GENERIC_CREDENTIALS, default_username, "", credbuf, credbufsize)
708 @assert succeeded
709
710 # Step 2: Create the actual dialog
711 # 2.1: Set up the window
712 messageArr = Base.cwstring(message)
713 captionArr = Base.cwstring(caption)
714 pfSave = Ref{Bool}(false)
715 cred = Ref{CREDUI_INFO}(CREDUI_INFO(sizeof(CREDUI_INFO), C_NULL, pointer(messageArr), pointer(captionArr), C_NULL))
716 dwflags = CREDUIWIN_GENERIC | CREDUIWIN_ENUMERATE_CURRENT_USER
717 if !prompt_username
718 # Disable setting anything other than default_username
719 dwflags |= CREDUIWIN_IN_CRED_ONLY
720 end
721 authPackage = Ref{Culong}(0)
722 outbuf_data = Ref{Ptr{Void}}(C_NULL)
723 outbuf_size = Ref{Culong}(0)
724
725 # 2.2: Do the actual request
726 code = ccall((:CredUIPromptForWindowsCredentialsW, "credui.dll"), stdcall, UInt32, (Ptr{CREDUI_INFO}, UInt32, Ptr{Culong},
727 Ptr{Void}, Culong, Ptr{Ptr{Void}}, Ptr{Culong}, Ptr{Bool}, UInt32), cred, 0, authPackage, credbuf, credbufsize[],
728 outbuf_data, outbuf_size, pfSave, dwflags)
729
730 # 2.3: If that failed for any reason other than the user canceling, error out.
731 # If the user canceled, just return a nullable
732 if code == ERROR_CANCELLED
733 return Nullable{Tuple{String,String}}()
734 elseif code != ERROR_SUCCESS
735 error(Base.Libc.FormatMessage(code))
736 end
737
738 # Step 3: Convert encrypted credentials back to plain text
739 passbuf = Array{UInt16,1}(1024)
740 passlen = Ref{UInt32}(length(passbuf))
741 usernamebuf = Array{UInt16,1}(1024)
742 usernamelen = Ref{UInt32}(length(usernamebuf))
743 # Need valid buffers for domain, even though we don't care
744 dummybuf = Array{UInt16,1}(1024)
745 succeeded = ccall((:CredUnPackAuthenticationBufferW, "credui.dll"), Bool,
746 (UInt32, Ptr{Void}, UInt32, Ptr{UInt16}, Ptr{UInt32}, Ptr{UInt16}, Ptr{UInt32}, Ptr{UInt16}, Ptr{UInt32}),
747 0, outbuf_data[], outbuf_size[], usernamebuf, usernamelen, dummybuf, Ref{UInt32}(1024), passbuf, passlen)
748 if !succeeded
749 error(Base.Libc.FormatMessage())
750 end
751
752 # Step 4: Free the encrypted buffer
753 # ccall(:SecureZeroMemory, Ptr{Void}, (Ptr{Void}, Csize_t), outbuf_data[], outbuf_size[]) - not an actual function
754 unsafe_securezero!(outbuf_data[], outbuf_size[])
755 ccall((:CoTaskMemFree, "ole32.dll"), Void, (Ptr{Void},), outbuf_data[])
756
757 # Done.
758 passbuf_ = passbuf[1:passlen[]-1]
759 result = Nullable((String(transcode(UInt8, usernamebuf[1:usernamelen[]-1])),
760 String(transcode(UInt8, passbuf_))))
761 securezero!(passbuf_)
762 securezero!(passbuf)
763
764 return result
765 end

780 """
781 @kwdef typedef
782
783 This is a helper macro that automatically defines a keyword-based constructor for the type
784 declared in the expression `typedef`, which must be a `struct` or `mutable struct`
785 expression. The default argument is supplied by declaring fields of the form `field::T =
786 default`. If no default is provided then the default is provided by the `kwdef_val(T)`
787 function.
788
789 ```julia
790 @kwdef struct Foo
791 a::Cint # implied default Cint(0)
792 b::Cint = 1 # specified default
793 z::Cstring # implied default Cstring(C_NULL)
794 y::Bar # implied default Bar()
795 end
796 ```
797 """
798 macro kwdef(expr)
799 expr = macroexpand(expr) # to expand @static
800 T = expr.args[2]
801 params_ex = Expr(:parameters)
802 call_ex = Expr(:call, T)
803 _kwdef!(expr.args[3], params_ex, call_ex)
804 quote
805 Base.@__doc__($(esc(expr)))
806 $(esc(Expr(:call,T,params_ex))) = $(esc(call_ex))
807 end
808 end
809
810 # @kwdef helper function
811 # mutates arguments inplace
812 function _kwdef!(blk, params_ex, call_ex)
813 for i in eachindex(blk.args)
814 ei = blk.args[i]
815 isa(ei, Expr) || continue
816 if ei.head == :(=)
817 # var::Typ = defexpr
818 dec = ei.args[1] # var::Typ
819 var = dec.args[1] # var
820 def = ei.args[2] # defexpr
821 push!(params_ex.args, Expr(:kw, var, def))
822 push!(call_ex.args, var)
823 blk.args[i] = dec
824 elseif ei.head == :(::)
825 dec = ei # var::Typ
826 var = dec.args[1] # var
827 def = :(Base.kwdef_val($(ei.args[2])))
828 push!(params_ex.args, Expr(:kw, var, def))
829 push!(call_ex.args, dec.args[1])
830 elseif ei.head == :block
831 # can arise with use of @static inside type decl
832 _kwdef!(ei, params_ex, call_ex)
833 end
834 end
835 blk
836 end

846 - zero for integer types
